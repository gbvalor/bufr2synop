29 \brief Parse a expanded descriptor with X = 31
30 \param syn pointer to a struct \ref synop_chunks where to set the results
31 \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
32
33 It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
34*/
35int syn_parse_x31 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
36{
37 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
38 {
39 return 0;
40 }
41
42 if ( syn == NULL )
43 {
44 return 1;
45 }
46
47 switch ( s->a->desc.y )
48 {
49 case 1:
50 s->rep = s->ival;
51 s->k_rep = s->i;
52 break;
53 default:
54 break;
55 }
56 return 0;
57}

92/*!
93 \fn int temp_parse_x31 ( struct temp_chunks *t, struct bufr2tac_subset_state *s )
94 \brief Parse a expanded descriptor with X = 31
95 \param t pointer to a struct \ref temp_chunks where to set the results
96 \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
97
98 It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
99*/
100int temp_parse_x31 ( struct temp_chunks *t, struct bufr2tac_subset_state *s )
101{
102 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
103 {
104 return 0;
105 }
106
107 if ( t == NULL )
108 {
109 return 1;
110 }
111
112 switch ( s->a->desc.y )
113 {
114 case 1: // 0 31 001 . Replicator
115 // It is supposed that it is the extended replicator used when describing
116 // wind shear points in raiosonde
117 // It is the amount of points of wind shear data at pressure level
118 if ( s->ival < TEMP_NMAX_POINTS )
119 {
120 s->itval = s->ival;
121 s->rep = 0; // used to mark it is a share point in sequent descriptors
122 }
123 else
124 {
125 return 1; // too much points
126 }
127 s->k_itval = s->i;
128 s->w->n = 0;
129 break;
130
131 case 2: // 0 31 002 . Extended replicator
132 // It is supposed that it is the extended replicator used when describing
133 // Temperature, dew-point and wind data at a pressure level with radiosonde position
134 // So the integer value of repliactor IS the amount of points
135 s->rep = s->ival; // replications and points we need. Also Used to mark this type of point
136 s->itval = 0;
137 s->k_rep = s->i;
138 s->r->n = 0;
139 break;
140 default:
141 break;
142 }
143 return 0;
144}
