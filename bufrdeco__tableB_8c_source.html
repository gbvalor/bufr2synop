34/*!
35 \fn int bufr_read_tableB ( struct bufrdeco b )
36 \brief Read a Table B file from a WMO csv formatted file and set the result in a struct \ref bufr_tableB
37 \param b pointer to the struct \ref bufrdeco where to set the results
38
39 Note that this function assumes that file is csv formatted as WMO table B
40 Return 0 if success, 1 otherwise
41*/
42int bufr_read_tableB ( struct bufrdeco *b )
43{
44 char *c;
45 FILE *t;
46 buf_t i = 0;
47 int nt;
48 uint32_t ix;
49 char l[CSV_MAXL];
50 char caux[512];
51 char *tk[16];
52 struct bufr_tableB *tb;
53 struct bufr_descriptor desc;
54
55 //bufrdeco_assert ( b != NULL );
56
57 tb = &(b->tables->b);
58 if ( tb->path[0] == 0 )
59 {
60 return 1;
61 }
62
63 // If we've already readed this table. We just regenerate the table with original values
64 if ( strcmp ( tb->path, tb->old_path ) == 0 )
65 {
66#ifdef __DEBUG
67 printf ("# Reused table %s\n", tb->path);
68#endif
69 for ( i = 0; i < tb->nlines ; i++ )
70 {
71 tb->item[i].scale = tb->item[i].scale_ori;
72 tb->item[i].reference = tb->item[i].reference_ori;
73 tb->item[i].nbits = tb->item[i].nbits_ori;
74 tb->item[i].changed = 0;
75 }
76 return 0; // all done
77 }
78
79 strcpy ( caux, tb->path );
80 memset ( tb, 0, sizeof ( struct bufr_tableB ) );
81 strcpy ( tb->path, caux );
82 if ( ( t = fopen ( tb->path, "r" ) ) == NULL )
83 {
84 snprintf ( b->error, sizeof (b->error),"Unable to open table B file '%s'\n", tb->path );
85 return 1;
86 }
87
88 // read first line, it is ignored
89 fgets ( l, CSV_MAXL, t );
90
91 while ( fgets ( l, CSV_MAXL, t ) != NULL && i < BUFR_MAXLINES_TABLEB )
92 {
93 // Parse line
94 if ( parse_csv_line ( &nt, tk, l ) < 0 || nt != 7 )
95 {
96 snprintf ( b->error, sizeof (b->error),"Error parsing csv line from table B file '%s'\n", tb->path );
97 return 1;
98 }
99
100 // First we build the descriptor
101 ix = strtoul ( tk[0], &c, 10 );
102 uint32_t_to_descriptor ( &desc, ix );
103 tb->item[i].changed = 0; // Original from table B
104 tb->item[i].x = desc.x; // x
105 tb->item[i].y = desc.y; // y
106 strcpy ( tb->item[i].key, desc.c ); // key
107 if ( tb->num[desc.x] == 0 )
108 {
109 tb->x_start[desc.x] = i; // marc the start
110 }
111 tb->y_ref[desc.x][desc.y] = i - tb->x_start[desc.x]; // marc the position from start of first x
112 ( tb->num[desc.x] )++;
113
114 // detailed name
115 if ( tk[2][0] )
116 {
117 sprintf ( caux,"%s %s", tk[1], tk[2] );
118 }
119 else
120 {
121 strcpy ( caux, tk[1] );
122 }
123
124 // and finally copy
125 strcpy ( tb->item[i].name, caux );
126
127 // unit
128 strcpy ( caux, tk[3] );
129 strcpy ( tb->item[i].unit, caux );
130
131 // escale
132 tb->item[i].scale_ori = strtol ( tk[4], &c, 10 );
133 tb->item[i].scale = tb->item[i].scale_ori;
134
135 // reference
136 tb->item[i].reference_ori = strtol ( tk[5], &c, 10 );
137 tb->item[i].reference = tb->item[i].reference_ori;
138
139 // bits
140 tb->item[i].nbits_ori = strtol ( tk[6], &c, 10 );
141 tb->item[i].nbits = tb->item[i].nbits_ori;
142
143 i++;
144 }
145 tb->x_start[0] = 0; // fix the start for x = 0
146 fclose ( t );
147 tb->nlines = i;
148 tb->wmo_table = 1;
149 strcpy ( tb->old_path, tb->path ); // store latest path
150 return 0;
151}

154/*!
155 \fn int bufr_restore_original_tableB_item ( struct bufr_tableB *tb, struct bufrdeco *b, uint8_t mode, char *key )
156 \brief Restores the original table B parameters for a BUFR descriptor
157 \param tb pointer to struct \ref bufr_tableB where are stored all table B data
158 \param b pointer to the basic struct \ref bufrdeco
159 \param mode integer with bit mask about changed parameteres by operator descriptors
160 \param key descriptor string in format FXXYYY
161
162 Return 0 if success, 1 otherwise
163*/
164int bufr_restore_original_tableB_item ( struct bufr_tableB *tb, struct bufrdeco *b, uint8_t mode, char *key )
165{
166 buf_t i;
167
168 bufrdeco_assert ( b != NULL && tb != NULL );
169
170 if ( bufr_find_tableB_index ( &i, tb, key ) )
171 {
172 snprintf (b->error, sizeof (b->error), "%s(): descriptor '%s' not found in table B\n", __func__, key );
173 return 1; // descritor not found
174 }
175
176 // escale
177 if ( mode & BUFR_TABLEB_CHANGED_SCALE || mode == 0 )
178 {
179 tb->item[i].scale = tb->item[i].scale_ori;
180 tb->item[i].changed &= ~ ( BUFR_TABLEB_CHANGED_SCALE ); // Clear bit mask
181 }
182
183 // reference
184 if ( mode & BUFR_TABLEB_CHANGED_REFERENCE || mode == 0 )
185 {
186 tb->item[i].reference = tb->item[i].reference_ori;
187 tb->item[i].changed &= ~ ( BUFR_TABLEB_CHANGED_REFERENCE ); // Clear bit mask
188 }
189 // bits
190 if ( mode & BUFR_TABLEB_CHANGED_BITS || mode == 0 )
191 {
192 tb->item[i].nbits = tb->item[i].nbits_ori;
193 tb->item[i].changed &= ~ ( BUFR_TABLEB_CHANGED_BITS ); // Clear bit mask
194 }
195 return 0;
196}

198/*!
199 \fn int bufr_find_tableB_index ( buf_t *index, struct bufr_tableB *tb, const char *key )
200 \brief found a descriptor index in a struct \ref bufr_tableB
201 \param index pointer to a size_t where to set the result if success
202 \param tb pointer to struct \ref bufr_tableB where are stored all table B data
203 \param key descriptor string in format FXXYYY
204
205 Return 0 if success, 1 otherwise
206*/
207int bufr_find_tableB_index ( buf_t *index, struct bufr_tableB *tb, const char *key )
208{
209 uint32_t ix;
210 buf_t i, i0;
211 char *c;
212 struct bufr_descriptor desc;
213
214 //bufrdeco_assert ( tb != NULL);
215
216 ix = strtoul ( key, &c, 10 );
217 uint32_t_to_descriptor ( &desc, ix );
218
219 // first chance
220 i0 = tb->x_start[desc.x] + tb->y_ref[desc.x][desc.y];
221 if ( tb->item[i0].x == desc.x && tb->item[i0].y == desc.y )
222 {
223 *index = i0;
224 return 0;
225 }
226 // Second chance
227 i0 = tb->x_start[desc.x];
228 for ( i = i0 ; i < i0 + tb->num[desc.x] ; i++ )
229 {
230 if ( tb->item[i].x != desc.x )
231 {
232 return 1; // descriptor not found
233 }
234 if ( tb->item[i].y == desc.y )
235 {
236 *index = i;
237 return 0; // found
238 }
239 }
240 return 1; // not found
241}

243/*!
244 \fn int bufrdeco_tableB_compressed ( struct bufrdeco_compressed_ref *r, struct bufrdeco *b, struct bufr_descriptor *d, int mode )
245 \brief get data from table B when parsing compressed data references
246 \param r pointer to a struct \ref bufrdeco_compressed_ref where to set results
247 \param b basic container struct \ref bufrdeco
248 \param d pointer to the reference struct \ref bufr_descriptor
249 \param mode If 0 then we are getting the data itself for a descriptor. If mode = 1 then we are dealing with associated bits
250
251 If succeeded returns 0. If problem returns 1. If mode = 1 and no associated bits returns -1
252*/
253int bufrdeco_tableB_compressed ( struct bufrdeco_compressed_ref *r, struct bufrdeco *b, struct bufr_descriptor *d, int mode )
254{
255 buf_t i;
256 uint32_t ival;
257 uint8_t has_data;
258 struct bufr_tableB *tb;
259
260 bufrdeco_assert ( b != NULL && r != NULL && d != NULL );
261
262 tb = & ( b->tables->b );
263
264 // if mode 1 then we search for associated_bits
265 if ( mode && b->state.assoc_bits == 0 )
266 {
267 // This is not an associated field, return -1
268 return -1;
269 }
270
271 if ( mode )
272 {
273 r->is_associated = 1; // Mark this reference as asociated
274 }
275
276 if ( is_a_local_descriptor ( d ) )
277 {
278 // if is a local descriptor we just skip the needed bits signified by operator 2 06 YYY
279 // we assign ref = 0 and ref0 as readed reserved bits. We also will assume escale = 0
280 memcpy ( & ( r->desc ), d, sizeof ( struct bufr_descriptor ) );
281 r->bits = b->state.local_bit_reserved;
282 r->bit0 = b->state.bit_offset; // OFFSET
283 strcpy_safe ( r->name, "LOCAL DESCRIPTOR" );
284 strcpy_safe ( r->unit, "UNKNOWN" );
285
286 // get bits for ref0
287 if ( get_bits_as_uint32_t ( &r->ref0, &r->has_data, &b->sec4.raw[4], & ( b->state.bit_offset ),
288 b->state.local_bit_reserved ) == 0 )
289 {
290 snprintf (b->error, sizeof (b->error), "%s(): Cannot get bits from '%s'\n", __func__, d->c );
291 return 1;
292 }
293
294 // and get 6 bits for inc_bits
295 if ( get_bits_as_uint32_t ( &ival, &has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), 6 ) == 0 )
296 {
297 snprintf (b->error, sizeof (b->error), "%s(): Cannot get 6 bits for inc_bits from '%s'\n", __func__, d->c );
298 return 1;
299 }
300
301 r->escale = 0;
302 r->inc_bits = ( uint8_t ) ival; // set the inc_bits
303 // Update the bit offset
304 b->state.bit_offset += r->inc_bits * b->sec3.subsets;
305 b->state.local_bit_reserved = 0; // Clean the reserved bits
306
307 // All is done for this local descriptor
308 return 0;
309 }
310
311 // build the index for tableB
312 i = tb->x_start[d->x] + tb->y_ref[d->x][d->y];
313
314 // copy the descriptor to reference member desc
315 memcpy ( & ( r->desc ), d, sizeof ( struct bufr_descriptor ) );
316 r->ref = tb->item[i].reference_ori; // copy the reference value from tableB, first from original
317 r->bits = tb->item[i].nbits + b->state.added_bit_length; // copy the bits from tableB
318 r->escale = tb->item[i].scale; // copy the scale from Tableb
319 strcpy_safe ( r->name, tb->item[i].name ); // copy the name
320 strcpy_safe ( r->unit, tb->item[i].unit ); // copy the unit name
321
322 // Add the added_scaled if not flag or code table
323 if ( strstr ( r->unit, "CODE TABLE" ) != r->unit && strstr ( r->unit,"FLAG" ) != r->unit &&
324 strstr ( r->unit, "Code table" ) != r->unit && strstr ( r->unit,"Flag" ) != r->unit )
325 {
326 r->escale += b->state.added_scale;
327 }
328 r->bit0 = b->state.bit_offset; // Sets the reference offset to current state offset
329 r->cref0[0] = '\0'; // default
330 r->ref0 = 0 ; // default
331
332 // check if is changing reference then we change reference value
333 if ( b->state.changing_reference != 255 )
334 {
335 // The descriptor operator 2 03 YYY is on action
336 // get the bits
337 if ( get_bits_as_uint32_t ( &ival, &r->has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), b->state.changing_reference ) == 0 )
338 {
339 snprintf (b->error, sizeof (b->error), "%s(): Cannot get bits from '%s'\n", __func__, d->c );
340 return 1;
341 }
342
343 // Change the new reference value in tableB with value previously readed because of rules for negative numbers
344 // (we still have the original value in reference_ori)
345 if ( get_table_b_reference_from_uint32_t ( & ( tb->item[i].reference ), b->state.changing_reference, ival ) )
346 {
347 snprintf (b->error, sizeof (b->error), "%s(): Cannot change reference in 2 03 YYY operator for '%s'\n", __func__, d->c );
348 return 1;
349 }
350
351 strcpy_safe ( r->unit, "NEW REFERENCE" );
352 r->ref = tb->item[i].reference;
353
354 // extracting inc_bits from next 6 bits
355 if ( get_bits_as_uint32_t ( &ival, &has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), 6 ) == 0 )
356 {
357 snprintf (b->error, sizeof (b->error), "%s(): Cannot get 6 bits for inc_bits from '%s'\n", __func__, d->c );
358 return 1;
359 }
360 // Here is supossed that all subsets will have the same reference change,
361 // so inc_bits must be 0 and no inc data should be present
362 if ( ival )
363 {
364 snprintf (b->error, sizeof (b->error), "%s(): Bad format for compressed data when changing reference from '%s'\n", __func__, d->c );
365 return 1;
366 }
367 r->ref0 = 0; // here we also assume that ref0 = 0
368 r->inc_bits = 0;
369 return 0;
370 }
371
372 if ( strstr ( r->unit, "CCITT" ) != NULL )
373 {
374 // Case of CCITT string as unit
375
376 if ( b->state.fixed_ccitt != 0 ) // can be changed by 2 08 YYY operator
377 r->bits = 8 * b->state.fixed_ccitt;
378
379 if ( get_bits_as_char_array ( r->cref0, &r->has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), r->bits ) == 0 )
380 {
381 snprintf (b->error, sizeof (b->error), "%s(): Cannot get uchars from '%s'\n", __func__, d->c );
382 return 1;
383 }
384 // extracting inc_bits from next 6 bits
385 if ( get_bits_as_uint32_t ( &ival, &has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), 6 ) == 0 )
386 {
387 snprintf (b->error, sizeof (b->error), "%s(): Cannot get 6 bits for inc_bits from '%s'\n", __func__, d->c );
388 return 1;
389 }
390 r->inc_bits = ival;
391 r->ref0 = 0;
392
393 // Set the bit_offset after all bits needed by this element in all subsets
394 b->state.bit_offset += r->inc_bits * 8 * b->sec3.subsets;
395
396 // all is done here !!
397 return 0;
398 }
399
400 // is a numeric field, i.e, a data value, a flag code or a code
401 // get reference value
402 if ( mode ) // case of associated field
403 {
404 if ( get_bits_as_uint32_t ( &r->ref0, &r->has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), b->state.assoc_bits ) == 0 )
405 {
406 snprintf (b->error, sizeof (b->error), "%s(): Cannot get associated bits from '%s'\n", __func__, d->c );
407 return 1;
408 }
409 // patch for delayed descriptor: it allways have data
410 if ( is_a_delayed_descriptor ( d ) || is_a_short_delayed_descriptor( d ) )
411 r->has_data = 1;
412 }
413 else // case of data
414 {
415 if ( get_bits_as_uint32_t ( &r->ref0, &r->has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), r->bits ) == 0 )
416 {
417 snprintf (b->error, sizeof (b->error), "%s(): Cannot get the data bits from '%s'\n", __func__, d->c );
418 return 1;
419 }
420 // patch for delayed descriptor: it allways have data
421 if ( is_a_delayed_descriptor ( d ) || is_a_short_delayed_descriptor ( d ) )
422 r->has_data = 1;
423 }
424
425 // extracting inc_bits from next 6 bits for inc_bits
426 if ( get_bits_as_uint32_t ( &ival, &has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), 6 ) == 0 )
427 {
428 snprintf (b->error, sizeof (b->error), "%s(): Cannot get 6 bits for inc_bits from '%s'\n", __func__, d->c );
429 return 1;
430 }
431 r->inc_bits = ival;
432
433 // if is a delayed descriptor then inc_bits MUST be 0.
434 if ( (is_a_delayed_descriptor ( d ) || is_a_short_delayed_descriptor( d ) ) && r->inc_bits )
435 {
436 snprintf (b->error, sizeof (b->error), "%s(): Found a delayed descriptor with inc_bits != 0\n", __func__ );
437 return 1;
438 }
439
440 // Set the bit_offset after all bits needed by this element in all subsets
441 b->state.bit_offset += r->inc_bits * b->sec3.subsets;
442
443 return 0;
444}

446/*!
447 \fn int bufrdeco_tableB_val ( struct bufr_atom_data *a, struct bufrdeco *b, struct bufr_descriptor *d )
448 \brief Get data from a table B descriptor
449 \param a pointer to a struct \ref bufr_atom_data where to set the results
450 \param b pointer to the basic struct \ref bufrdeco
451 \param d pointer to the target descriptor
452
453 Return 0 if success, 1 otherwise
454*/
455int bufrdeco_tableB_val ( struct bufr_atom_data *a, struct bufrdeco *b, struct bufr_descriptor *d )
456{
457 buf_t i, nbits = 0;
458 uint32_t ival;
459 uint8_t has_data;
460 int32_t /*escale = 0,*/ reference = 0;
461 struct bufr_tableB *tb;
462
463 //bufrdeco_assert ( a != NULL && b != NULL && d != NULL );
464
465 tb = & ( b->tables->b );
466
467 if ( is_a_local_descriptor ( d ) )
468 {
469 // if is a local descriptor we just skip the bits signified by operator 2 06 YYY
470 a->mask = DESCRIPTOR_IS_LOCAL;
471 strcpy_safe ( a->name, "LOCAL DESCRIPTOR" );
472 strcpy_safe ( a->unit, "UNKNOWN" );
473 if ( get_bits_as_uint32_t ( &ival, &has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), b->state.local_bit_reserved ) == 0 )
474 {
475 snprintf (b->error, sizeof (b->error), "%s(): Cannot get bits from '%s'\n", __func__, d->c );
476 return 1;
477 }
478 a->val = ival; // we assume escale = 0 and ref = 0
479 b->state.local_bit_reserved = 0; // Clean the reserved bits
480 return 0;
481 }
482
483 i = tb->x_start[d->x] + tb->y_ref[d->x][d->y];
484
485 memcpy ( & ( a->desc ), d, sizeof ( struct bufr_descriptor ) );
486 a->mask = 0;
487 strcpy_safe ( a->name, tb->item[i].name );
488 strcpy_safe ( a->unit, tb->item[i].unit );
489 a->escale = tb->item[i].scale;
490
491 // Case of difference statistics active
492 if ( b->state.dstat_active )
493 nbits = tb->item[i].nbits + 1;
494 else
495 nbits = tb->item[i].nbits;
496
497 if ( b->state.changing_reference != 255 )
498 {
499 // The descriptor operator 2 03 YYY is on action
500 // Get preliminar value
501 if ( get_bits_as_uint32_t ( &ival, &has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), b->state.changing_reference ) == 0 )
502 {
503 snprintf (b->error, sizeof (b->error), "%s(): Cannot get bits from '%s'\n", __func__, d->c );
504 return 1;
505 }
506
507 // Then change the preliminar reference value because of rules for negative numbers
508 if ( get_table_b_reference_from_uint32_t ( & ( tb->item[i].reference ), b->state.changing_reference, ival ) )
509 {
510 snprintf (b->error, sizeof (b->error), "%s(): Cannot change reference in 2 03 YYY operator for '%s'\n", __func__, d->c );
511 return 1;
512 }
513 strcpy_safe ( a->unit, "NEW REFERENCE" );
514 a->val = ( double ) tb->item[i].reference;
515 return 0;
516 }
517
518 // case of difference statistics active
519 if ( b->state.dstat_active )
520 reference = - ( ( int32_t ) 1 << ( tb->item[i].nbits ) );
521 else
522 reference = tb->item[i].reference;
523
524 //printf(" escale = %d reference = %d nbits = %lu\n", escale, reference, nbits);
525 if ( strstr ( a->unit, "CCITT" ) != NULL )
526 {
527 if ( b->state.fixed_ccitt != 0 ) // can be changed by 2 08 YYY operator
528 nbits = 8 * b->state.fixed_ccitt;
529
530 if ( get_bits_as_char_array ( a->cval, &has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), nbits ) == 0 )
531 {
532 snprintf (b->error, sizeof (b->error), "%s(): Cannot get uchars from '%s'\n", __func__, d->c );
533 return 1;
534 }
535 if ( has_data == 0 )
536 {
537 a->mask |= DESCRIPTOR_VALUE_MISSING;
538 }
539 else
540 {
541 a->mask |= DESCRIPTOR_HAVE_STRING_VALUE;
542 }
543 return 0;
544 }
545
546 // is a numeric field, i.e, a data value, a flag code or a code
547 // Set associated bits
548 if ( b->state.assoc_bits &&
549 a->desc.x != 31 && // Data description qualifier has not associated bits itself
550 get_bits_as_uint32_t ( &a->associated, &has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), b->state.assoc_bits ) == 0 )
551 {
552 snprintf (b->error, sizeof (b->error), "%s(): Cannot get associated bits from '%s'\n", __func__, d->c );
553 return 1;
554 }
555 else
556 {
557 a->associated = MISSING_INTEGER;
558 }
559
560 if ( strstr ( a->unit, "CODE TABLE" ) != a->unit && strstr ( a->unit,"FLAG" ) != a->unit &&
561 strstr ( a->unit, "Code table" ) != a->unit && strstr ( a->unit,"Flag" ) != a->unit )
562 {
563 // case of numeric, no string nor code nor flag
564 nbits += b->state.added_bit_length;
565 }
566
567 if ( get_bits_as_uint32_t ( &ival, &has_data, &b->sec4.raw[4], & ( b->state.bit_offset ), nbits ) == 0 )
568 {
569 snprintf (b->error, sizeof (b->error), "%s(): Cannot get bits from '%s'\n", __func__, d->c );
570 return 1;
571 }
572
573 // patch for delayed descriptor: it allways have data
574 if ( a->desc.x == 31 )
575 has_data = 1;
576
577 if ( has_data )
578 {
579 if ( strstr ( a->unit, "CODE TABLE" ) != a->unit && strstr ( a->unit,"FLAG" ) != a->unit &&
580 strstr ( a->unit, "Code table" ) != a->unit && strstr ( a->unit,"Flag" ) != a->unit )
581 {
582 a->escale += b->state.added_scale;
583 reference += b->state.added_reference;
584 if ( b->state.factor_reference > 1 )
585 reference *= b->state.factor_reference;
586 }
587 // Get a numeric number
588 if ( a->escale >= 0 && a->escale < 8 )
589 {
590 a->val = ( double ) ( ( int32_t ) ival + reference ) * pow10neg[ ( size_t ) a->escale];
591 }
592 else if ( a->escale < 0 && a->escale > -8 )
593 {

599 }

602 {

604 a->mask |= DESCRIPTOR_IS_CODE_TABLE;

608 }
609 }
610 else if ( strstr ( a->unit,"FLAG" ) == a->unit || strstr ( a->unit,"Flag" ) == a->unit )

618 }
619 }
620 }
621 else
622 {
623 a->val = MISSING_REAL;
624 a->mask |= DESCRIPTOR_VALUE_MISSING;
625 }
626
627 return 0;
