242/*!
243 \fn int bufrdeco_fprint_tree_recursive ( FILE *f, struct bufrdeco *b, struct bufr_sequence *seq )
244 \brief Print a tree of descriptors to a file in a recursive way
245 \param f Pointer to file opened by caller
246 \param b pointer to the basic container struct \ref bufrdeco
247 \param seq pointer to the struct \ref bufr_sequence to print
248
249 If succeded return 0, else return 1
250*/
251int bufrdeco_fprint_tree_recursive ( FILE *f, struct bufrdeco *b, struct bufr_sequence *seq )
252{
253 buf_t i, j, k;
254 struct bufr_sequence *l;
255 char explanation[256];
256
257 bufrdeco_assert ( f != NULL && b != NULL );
258
259 if ( seq == NULL )
260 {
261 l = & ( b->tree->seq[0] );
262 }
263 else
264 {
265 l = seq;
266 }
267
268 for ( i = 0; i < l->ndesc; i++ )
269 {
270 // we search for descriptors woth f == 3
271
272 for ( j = 0; j < l->level; j++ )
273 {
274 fprintf ( f, " |" );
275 }
276 fprintf ( f, "%u %02u %03u", l->lseq[i].f, l->lseq[i].x,l->lseq[i].y );
277
278 if ( l->lseq[i].f != 3 )
279 {
280 if ( l->lseq[i].f == 0 )
281 {
282 if ( bufr_find_tableB_index ( &k, & ( b->tables->b ), l->lseq[i].c ) )
283 fprintf ( f, "\n" );
284 else
285 {
286 fprintf ( f, ":" );
287 if ( l->replicated[i] )
288 {
289 //fprintf ( f, " " );
290 for ( j = 0; j < l->replicated[i] ; j++ )
291 fprintf ( f, "*" );
292 }
293 if ( is_a_delayed_descriptor ( & l->lseq[i] ) ||
294 is_a_short_delayed_descriptor ( & l->lseq[i] ) )
295 fprintf ( f, "* %s\n", b->tables->b.item[k].name );
296 else
297 fprintf ( f, " %s\n", b->tables->b.item[k].name );
298 }
299 }
300 else if ( l->lseq[i].f == 2 )
301 {
302 fprintf ( f, ": %s\n", bufrdeco_get_f2_descriptor_explanation ( explanation, sizeof ( explanation ), & ( l->lseq[i] ) ) );
303 }
304 else if ( l->lseq[i].f == 1 )
305 {
306 if ( l->lseq[i].y == 0 )
307 {
308 if ( l->lseq[i].x == 1 )
309 fprintf ( f, ":* Replicator for %d descriptor after next delayed descriptor which set the number of replications.\n", l->lseq[i].x );
310 else
311 fprintf ( f, ":* Replicator for %d descriptors after next delayed descriptor which set the number of replications.\n", l->lseq[i].x );
312 }
313 else
314 {
315 if ( l->lseq[i].x == 1 )
316 fprintf ( f, ":* Replicator for next %d descriptors %d times\n", l->lseq[i].x, l->lseq[i].y );
317 else
318 fprintf ( f, ":* Replicator for next %d descriptors %d times\n", l->lseq[i].x, l->lseq[i].y );
319 }
320 }
321 else
322 fprintf ( f, "\n" );
323 continue;
324 }
325
326 // we then recursively parse the son
327 if ( l->replicated[i] )
328 {
329 fprintf ( f, ":" );
330 for ( j = 0; j < l->replicated[i] ; j++ )
331 fprintf ( f, "*" );
332 }
333
334 fprintf ( f, "-> %s\n", l->sons[i]->name );
335 if ( bufrdeco_fprint_tree_recursive ( f, b, l->sons[i] ) )
336 {
337 return 1;
338 }
339
340 }
341
342 return 0;
343}

412/*!
413 \fn char * bufrdeco_print_atom_data ( char *target, size_t lmax, struct bufr_atom_data *a )
414 \brief print the data in a struct \ref bufr_atom_data to a string
415 \param target string where to print the result
416 \param lmax size of allocated string \a target
417 \param a pointer to struct ref \ref bufr_atom_data with data to print
418
419 Returns a pointer to result string
420*/
421char * bufrdeco_print_atom_data ( char *target, size_t lmax, struct bufr_atom_data *a )
422{
423 char aux[256];
424 size_t used = 0;
425 size_t nlimit, climit;
426
427 bufrdeco_assert ( a != NULL && target != NULL );
428
429 used += snprintf ( target + used, lmax - used, "%u %02u %03u ", a->desc.f, a->desc.x, a->desc.y );
430 strcpy_safe ( aux, a->name );
431 aux[64] = '\0';
432 used += snprintf ( target + used, lmax - used, "%-64s ", aux );
433 strcpy_safe ( aux, a->unit );
434 aux[20] = '\0';
435 used += snprintf ( target + used, lmax - used, "%-20s", aux );
436 if ( a->mask & DESCRIPTOR_VALUE_MISSING )
437 {
438 used += snprintf ( target + used, lmax - used, "%17s", "MISSING" );
439 }
440 else
441 {
442 if ( a->mask & DESCRIPTOR_HAVE_STRING_VALUE )
443 {
444 strcpy_safe ( aux, a->cval );
445 aux[64] = '\0';
446 used += snprintf ( target + used, lmax - used, " " );
447 used += snprintf ( target + used, lmax - used, "%s", aux );
448 }
449 else if ( a->mask & DESCRIPTOR_HAVE_CODE_TABLE_STRING
450 || strstr ( a->unit, "CODE TABLE" ) == a->unit
451 || strstr ( a->unit, "Code table" ) == a->unit )
452 {
453 strcpy_safe ( aux, a->ctable );
454 aux[64] = '\0';
455 used += snprintf ( target + used, lmax - used, "%17u ", ( uint32_t ) a->val );
456 used += snprintf ( target + used, lmax - used, "%s", aux );
457 }
458 else if ( a->mask & DESCRIPTOR_HAVE_FLAG_TABLE_STRING )
459 {
460 strcpy_safe ( aux, a->ctable );
461 aux[64] = '\0';
462 used += snprintf ( target + used, lmax - used, " 0x%08X ", ( uint32_t ) a->val );
463 used += snprintf ( target + used, lmax - used, "%s", aux );
464 }
465 else
466 {
467 used += snprintf ( target + used, lmax - used, "%s ", get_formatted_value_from_escale ( aux, sizeof ( aux ), a->escale, a->val ) );
468 }
469
470 }
471
472 // Now print remaining chars in a->name or a->ctable
473 nlimit = 64;
474 climit = 64;
475 while ( ( strlen ( a->name ) > nlimit && nlimit < BUFR_TABLEB_NAME_LENGTH ) ||
476 ( strlen ( a->ctable ) > climit && climit < BUFR_EXPLAINED_LENGTH ) )
477 {
478 aux[0] = 0;
479 if ( strlen ( a->name ) > nlimit && nlimit < BUFR_TABLEB_NAME_LENGTH )
480 strncpy ( aux, a->name + nlimit, 64 );
481 used += snprintf ( target + used, lmax - used, "\n %-64s", aux );
482 aux[0] = 0;
483 if ( strlen ( a->ctable ) > climit && climit < BUFR_EXPLAINED_LENGTH )
484 strncpy ( aux, a->ctable + climit, 64 );
485 used += snprintf ( target + used, lmax - used, " %s", aux );
486 nlimit += 64;
487 climit += 64;
488 }
489
490 if ( a->is_bitmaped_by != 0 )
491 used += snprintf ( target + used, lmax - used, " *IS BITMAPED BY #%u*", a->is_bitmaped_by );
492
493 if ( a->bitmap_to != 0 )
494 used += snprintf ( target + used, lmax - used, " *BITMAP TO #%u*", a->bitmap_to );
495
496 if ( a->related_to != 0 )
497 used += snprintf ( target + used, lmax - used, " *RELATED TO #%u*", a->related_to );
498
499 return target;
500}
