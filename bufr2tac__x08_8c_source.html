29/*!
30 \fn int syn_parse_x08 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
31 \brief Parse a expanded descriptor with X = 08
32 \param syn pointer to a struct \ref synop_chunks where to set the results
33 \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
34
35 It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
36*/
37int syn_parse_x08 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
38{
39 if ( syn == NULL )
40 {
41 return 1;
42 }
43
44 switch ( s->a->desc.y )
45 {
46 case 2: // 0 08 002 . Vertical significance (surface observations)
47 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
48 {
49 s->clayer = 0; // clean vertical significance;
50 return 0;
51 }
52
53 if ( s->ival == 21 || s->ival == 1 )
54 {
55 s->clayer = 1; // first cloud layer or first no CB layer
56 }
57 else if ( s->ival == 22 || s->ival == 2 )
58 {
59 s->clayer = 2; // second cloud layer or second no CB layer
60 }
61 else if ( s->ival == 23 || s->ival == 3 )
62 {
63 s->clayer = 3; // third cloud layer or third no CB layer
64 }
65 else if ( s->ival == 24 || s->ival == 4 )
66 {
67 s->clayer = 4; // fourth cloud layer or CB layer
68 }
69 else if ( s->ival == 10 )
70 {
71 s->clayer = -1; // base of cloud layer below level station and top over level station
72 }
73 else if ( s->ival == 11 )
74 {
75 s->clayer = -2; // base and top of cloud layer below level station
76 }
77 else if ( s->ival <= 9 && s->ival >= 7 )
78 {
79 s->clayer = s->ival;
80 }
81 else if ( s->ival == 5 || s->ival == 62 )
82 {
83 // 5 should be used for N = 9 and 62 for N = 0
84 s->clayer = s->ival;
85 }
86 break;
87
88 case 22: // 0 08 022 . Total number
89 case 23: // 0 08 023 . First-order statistics
90 case 24: // 0 08 024 . Difference statistics
91 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
92 {
93 return 0;
94 }
95 if ( s->isq )
96 {
97 s->isq = 0;
98 }
99 else
100 {
101 s->isq = 1;
102 }
103 break;
104
105 default:
106 break;
107 }
108 return 0;
109}

160{

162
163 if ( c == NULL )
164 {
165 return 1;
166 }
167
168 switch ( s->a->desc.y )
169 {
170 case 20: // 0 08 020 . Total number of missing entities (days)
171 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
172 {
173 return 0;
174 }
175 if ( s->is_normal == 0 )
176 {
177 switch ( s->isq_val )
178 {
179 // days of missing data
180 case 1: // pressure

260 break;
261 }
262 }
263 break;
264
265 case 22:
266 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )

268 return 0;

387 case 53: // 0 08 053 . Day of occurrence qualifier (= 0; on 1 day only) (= 1; on 2 or more days)
388 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
389 {
390 s->isq = 0;
391 s->isq_val = -1;
392 return 0;
393 }
394 s->isq = 1;
395 s->isq_val = s->ival;
396 break;
397
398 default:
399 break;
400 }
401
402 return 0;

405/*!
406 \fn int temp_parse_x08 ( struct temp_chunks *t, struct bufr2tac_subset_state *s )
407 \brief Parse a expanded descriptor with X = 08
408 \param t pointer to a struct \ref temp_chunks where to set the results
409 \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
410
411 It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
412*/
413int temp_parse_x08 ( struct temp_chunks *t, struct bufr2tac_subset_state *s )
414{
415 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
416 {
417 return 0;
418 }
419
420 if ( t == NULL )
421 {
422 return 1;
423 }
424
425 switch ( s->a->desc.y )
426 {
427 case 2: // 0 08 002.
428 // this is to mark of the begining or end of cloud surface data
429 if ( s->isq == 0 )
430 {
431 s->isq = 1;
432 }
433 else
434 {
435 s->isq = 0;
436 }
437 break;
438 case 21: // 0 08 021. Time significance
439 if ( s->ival != 18 )
440 {
441 return 1; // it should be 18 (launch date/time)
442 }
443 break;
444
445 case 42: // 0 08 042. Extended vertical sounding significance
446 if ( s->rep > 0 && s->r->n > 0 )
447 {
448 s->r->raw[s->r->n - 1].flags = s->ival;
449 }
450 else if ( s->w->n > 0 )
451 {
452 // wind shear case
453 s->w->raw[s->w->n - 1].flags = s->ival;
454 }
455 break;
456 default:
457 break;
458 }
459 return 0;
460}
