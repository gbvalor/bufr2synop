26/*!
27 \fn struct bufrdeco_subset_sequence_data * bufrdeco_get_subset_sequence_data(struct bufrdeco *b)
28 \brief Parse and get a struct \ref bufrdeco_subset_sequence_data
29 \param b basic container struct \ref bufrdeco
30
31 Note that if succeeded the counter to current subset index is increased. Remember that in case of non
32 compressed data we must to decode all previous subsets to get the desired one. In case of compressed
33 we can access directly to a the desired subset
34
35 If succeeded returns a pointer to the struct \ref bufrdeco_subset_sequence_data with the results for a
36 subset, otherwise returns NULL
37*/
38struct bufrdeco_subset_sequence_data * bufrdeco_get_subset_sequence_data ( struct bufrdeco *b )
39{
40 if ( bufrdeco_decode_data_subset ( & ( b->seq ), & ( b->refs ), b ) )
41 {
42 return NULL;
43 }
44 return & ( b->seq );
45}

48/*!
49 \fn int bufrdeco_decode_data_subset ( struct bufrdeco_subset_sequence_data *s, struct bufrdeco_compressed_data_references *r, struct bufrdeco *b )
50 \brief User interface to decode a BUFR subset
51 \param s pointer to the target struct \ref bufrdeco_subset_sequence_data
52 \param r pointer to the struct \ref bufrdeco_compressed_data_references
53 \param b pointer to the base struct \ref bufrdeco
54
55 Note that if succeeded the counter to current subset index is increased. Remember that in case of non
56 compressed data we must to decode all previous subsets to get the desired one. In case of compressed
57 we can access directly to a the desired subset
58
59 Return 0 in case of success, 1 otherwise
60*/
61int bufrdeco_decode_data_subset ( struct bufrdeco_subset_sequence_data *s, struct bufrdeco_compressed_data_references *r,
62 struct bufrdeco *b )
63{
64 // Check about parsed tree
65 if ( b->tree == NULL || b->tree->nseq == 0 )
66 {
67 sprintf ( b->error, "bufrdeco_subset_sequence_datad(): Try to parse compressed data without parsed tree\n" );
68 return 1;
69 }
70
71 // First we init the subset sequence data
72 if ( bufrdeco_clean_subset_sequence_data ( s ) )
73 {
74 return 1;
75 }
76
77 // also we clean the possible defined bitmaps in prior subsets
78 if ( bufrdeco_clean_bitmaps ( b ) )
79 {
80 return 1;
81 }
82
83 // Then we get the data from an already parsed descriptor tree
84 if ( b->sec3.compressed )
85 {
86 if ( r->nd == 0 /*r->refs == NULL || r->dim == 0 */ )
87 {
88 // case of compressed data and still not parsed
89 if ( bufrdeco_parse_compressed ( r, b ) )
90 {
91 return 1;
92 }
93 }
94 if ( bufr_decode_subset_data_compressed ( s, r, b ) )
95 {
96 return 1;
97 }
98 }
99 else
100 {
101 if ( bufrdeco_decode_subset_data_recursive ( s, NULL, b ) )
102 {
103 return 1;
104 }
105 }
106
107 // Finally we update the subset counter
108 ( b->state.subset ) ++;
109 return 0;
110}

113/*!
114 \fn int bufrdeco_increase_data_array ( struct bufrdeco_subset_sequence_data *s )
115 \brief doubles the allocated space for a struct \ref bufrdeco_subset_sequence_data whenever is posible
116 \param s pointer to source struct \ref bufrdeco_subset_sequence_data
117
118 The amount of data in a bufr must be huge. In a first moment, the dimension of a sequence of structs
119 \ref bufr_atom_data is \ref BUFR_NMAXSEQ but may be increased. This function task is try to double the
120 allocated dimension and reallocate it.
121
122 Return 0 when success, otherwise return 1 and the struct is unmodified
123*/
124int bufrdeco_increase_data_array ( struct bufrdeco_subset_sequence_data *s )
125{
126 if ( s->dim < ( BUFR_NMAXSEQ * 8 ) ) // check if reached the limit
127 {
128 if ( ( s->sequence = ( struct bufr_atom_data * ) realloc ( ( void * ) s->sequence,
129 s->dim * 2 * sizeof ( struct bufr_atom_data ) ) ) == NULL )
130 {
131 return 1;
132 }
133 else
134 {
135 s->dim *= 2;
136 return 0;
137 }
138 }
139 else
140 {
141 return 1;
142 }
143}

145/*!
146 \fn int bufrdeco_decode_subset_data_recursive ( struct bufrdeco_subset_sequence_data *s, struct bufr_sequence *l, struct bufrdeco *b )
147 \brief decode the data from a subset in a recursive way
148 \param s pointer to the target struct \ref bufrdeco_subset_sequence_data
149 \param l pointer to the source struct \ref bufr_sequence
150 \param b pointer to the base struct \ref bufrdeco
151
152 Return 0 in case of success, 1 otherwise
153*/
154int bufrdeco_decode_subset_data_recursive ( struct bufrdeco_subset_sequence_data *s, struct bufr_sequence *l, struct bufrdeco *b )
155{
156 size_t i, k;
157 struct bufr_sequence *seq;
158 struct bufr_replicator replicator;
159
160 // clean subset data
161 if ( l == NULL )
162 {
163 //memset ( s, 0, sizeof ( struct bufr_subset_sequence_data ) );
164 s->nd = 0;
165 s->ss = b->state.subset;
166 seq = & ( b->tree->seq[0] );
167 if ( b->state.subset == 0 )
168 {
169 b->state.bit_offset = 0;
170 }
171 // also reset reference and bits inc
172 b->state.added_bit_length = 0;
173 b->state.added_scale = 0;
174 b->state.added_reference = 0;
175 b->state.assoc_bits = 0;
176 b->state.changing_reference = 255;
177 b->state.fixed_ccitt = 0;
178 b->state.local_bit_reserved = 0;
179 b->state.factor_reference = 1;
180 b->state.quality_active = 0;
181 b->state.subs_active = 0;
182 b->state.retained_active = 0;
183 b->state.stat1_active = 0;
184 b->state.dstat_active = 0;
185 b->state.bitmaping = 0;
186 b->state.bitmap = NULL;
187 }
188 else
189 {
190 seq = l;
191 }
192
193 // loop for a sequence
194 for ( i = 0; i < seq->ndesc ; i++ )
195 {
196 switch ( seq->lseq[i].f )
197 {
198 case 0:
199 // Get data from table B
200 if ( bufrdeco_tableb_val ( & ( s->sequence[s->nd] ), b, & ( seq->lseq[i] ) ) )
201 {
202 return 1;
203 }
204
205 //case of first order statistics
206 if ( b->state.stat1_active &&
207 seq->lseq[i].x == 8 && seq->lseq[i].y == 23 )
208 {
209 k = b->bitmap.bmap[b->bitmap.nba - 1]->ns1; // index un stqts
210 b->bitmap.bmap[b->bitmap.nba - 1]->stat1_desc[k] = s->nd; // Set the value of statistical parameter
211 // update the number of quality variables for the bitmap
212 if ( k < BUFR_MAX_QUALITY_DATA )
213 ( b->bitmap.bmap[b->bitmap.nba - 1]->ns1 )++;
214 else
215 {
216 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence_compressed(): No more space for first order statistic vars in bitmap. Check BUFR_MAX_QUALITY_DATA\n" );
217 return 1;
218 }
219 }
220
221 //case of difference statistics
222 if ( b->state.dstat_active &&
223 seq->lseq[i].x == 8 && seq->lseq[i].y == 24 )
224 {
225 k = b->bitmap.bmap[b->bitmap.nba - 1]->nds; // index in stats
226 b->bitmap.bmap[b->bitmap.nba - 1]->dstat_desc[k] = s->nd; // Set the value of statistical parameter
227 // update the number of quality variables for the bitmap
228 if ( k < BUFR_MAX_QUALITY_DATA )
229 ( b->bitmap.bmap[b->bitmap.nba - 1]->nds )++;
230 else
231 {
232 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence_compressed(): No more space for difference statistic vars in bitmap. Check BUFR_MAX_QUALITY_DATA\n" );
233 return 1;
234 }
235 }
236
237
238
239 // Add info about common sequence to which bufr_atom_data belongs
240 s->sequence[s->nd].seq = seq;
241 s->sequence[s->nd].ns = i;
242
243 //bufr_print_atom_data_stdout(& ( s->sequence[s->nd] ));
244 if ( s->nd < ( s->dim - 1 ) )
245 {
246 ( s->nd ) ++;
247 }
248 else if ( bufrdeco_increase_data_array ( s ) == 0 )
249 {
250 ( s->nd ) ++;
251 }
252 else
253 {
254 sprintf ( b->error, "bufr_decode_data_subset_recursive(): No more bufr_atom_data available. Check BUFR_NMAXSEQ\n" );
255 return 1;
256 }
257 break;
258
259 case 1:
260 // Case of replicator descriptor
261 replicator.s = seq;
262 replicator.ixrep = i;
263 if ( seq->lseq[i].y != 0 )
264 {
265 // no delayed
266 replicator.ixdel = i;
267 replicator.ndesc = seq->lseq[i].x;
268 replicator.nloops = seq->lseq[i].y;
269
270 // Check if this replicator is for a bit-map defining
271 if ( b->state.bitmaping )
272 {
273 b->state.bitmaping = replicator.nloops; // set it properly
274 }
275
276
277 bufrdeco_decode_replicated_subsequence ( s, &replicator, b );
278
279 // and then set again bitamping to 0, because it is finished
280 b->state.bitmaping = 0;
281
282 i += replicator.ndesc;
283 }
284 else
285 {
286 // case of delayed;
287 replicator.ixdel = i + 1;
288 replicator.ndesc = seq->lseq[i].x;
289 // here we read ndesc from delayed replicator descriptor
290 if ( bufrdeco_tableb_val ( & ( s->sequence[s->nd] ), b, & ( seq->lseq[i + 1] ) ) )
291 {
292 return 1;
293 }
294 // Add info about common sequence to which bufr_atom_data belongs
295 s->sequence[s->nd].seq = seq;
296 s->sequence[s->nd].ns = i + 1;
297
298 replicator.nloops = ( size_t ) ( s->sequence[s->nd].val );
299
300 // Check if this replicator is for a bit-map defining
301 if ( b->state.bitmaping )
302 {
303 b->state.bitmaping = replicator.nloops; // set it properly
304 }
305
306 if ( s->nd < ( s->dim - 1 ) )
307 {
308 ( s->nd ) ++;
309 }
310 else if ( bufrdeco_increase_data_array ( s ) == 0 )
311 {
312 ( s->nd ) ++;
313 }
314 else
315 {
316
317 sprintf ( b->error, "bufr_decode_data_subset_recursive(): No more bufr_atom_data available. Check BUFR_NMAXSEQ\n" );
318 return 1;
319 }
320 bufrdeco_decode_replicated_subsequence ( s, &replicator, b );
321
322 // and then set again bitamping to 0, because it is finished
323 b->state.bitmaping = 0;
324
325 i += replicator.ndesc + 1;
326 }
327 //i = replicator.ixdel + replicator.ndesc; // update i properly
328 break;
329
330 case 2:
331 // Case of operator descriptor
332 if ( bufrdeco_parse_f2_descriptor ( s, & ( seq->lseq[i] ), b ) )
333 {
334 return 1;
335 }
336 break;
337
338 case 3:
339 // Case of sequence descriptor
340 if ( bufrdeco_decode_subset_data_recursive ( s, seq->sons[i], b ) )
341 {
342 return 1;
343 }
344 break;
345
346 default:
347 // this case is not possible
348 sprintf ( b->error, "bufr_decode_data_subset_recursive(): Found bad 'f' in descriptor\n" );
349 return 1;
350 break;
351 }
352 }
353 return 0;
354};

356/*!
357 \fn int bufrdeco_decode_replicated_subsequence ( struct bufrdeco_subset_sequence_data *s, struct bufr_replicator *r, struct bufrdeco *b )
358 \brief Decodes a replicated sequence
359 \param s target struct \ref bufrdeco_subset_sequence_data
360 \param r pointer to a struct \ref bufr_replicator which manage the replication task
361 \param b pointer to the base struct \ref bufrdeco
362
363 If succeeded return 0, 1 otherwise
364*/
365int bufrdeco_decode_replicated_subsequence ( struct bufrdeco_subset_sequence_data *s, struct bufr_replicator *r, struct bufrdeco *b )
366{
367 size_t i, k;
368 size_t ixloop; // Index for loop
369 size_t ixd; // Index for descriptor
370 struct bufr_sequence *l = r->s; // sequence
371 struct bufr_replicator replicator;
372
373 //printf("nloops=%lu, ndesc=%lu\n", r->nloops, r->ndesc);
374 for ( ixloop = 0; ixloop < r->nloops; ixloop++ )
375 {
376 for ( ixd = 0; ixd < r->ndesc ; ixd ++ )
377 {
378 i = ixd + r->ixdel + 1;
379 switch ( l->lseq[i].f )
380 {
381 case 0:
382 // Get data from table B
383 if ( bufrdeco_tableb_val ( & ( s->sequence[s->nd] ), b, & ( l->lseq[i] ) ) )
384 {
385 return 1;
386 }
387
388 // Case of defining bitmap 0 31 031
389 if ( l->lseq[i].x == 31 && l->lseq[i].y == 31 && b->state.bitmaping )
390 {
391 if ( s->sequence[s->nd].val == 0.0 ) // Check if it is meaning present data
392 {
393 s->sequence[s->nd - b->state.bitmaping].is_bitmaped_by = s->nd;
394 s->sequence[s->nd].bitmap_to = s->nd - b->state.bitmaping;
395 // Add reference to bitmap
396 bufrdeco_add_to_bitmap( b->bitmap.bmap[b->bitmap.nba - 1], s->nd - b->state.bitmaping, s->nd);
397 }
398 }
399
400 // Process quality data
401 if ( b->state.quality_active &&
402 l->lseq[i].x == 33 )
403 {
404 if ( ixloop == 0 )
405 {
406 k = b->bitmap.bmap[b->bitmap.nba - 1]->nq;
407 b->bitmap.bmap[b->bitmap.nba - 1]->quality[k] = s->nd;
408 if ( k < BUFR_MAX_QUALITY_DATA )
409 ( b->bitmap.bmap[b->bitmap.nba - 1]->nq )++;
410 else
411 {
412 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence(): No more space for quality vars in bitmap. Check BUFR_MAX_QUALITY_DATA\n" );
413 return 1;
414 }
415 }
416 s->sequence[s->nd].related_to = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop];
417 }
418
419 //case of first order statistics
420 if ( b->state.stat1_active &&
421 l->lseq[i].x == 8 && l->lseq[i].y == 23 )
422 {
423 if ( ixloop == 0 )
424 {
425 k = b->bitmap.bmap[b->bitmap.nba - 1]->ns1; // index un stqts
426 b->bitmap.bmap[b->bitmap.nba - 1]->stat1_desc[k] = s->nd; // Set the value of statistical parameter
427 // update the number of quality variables for the bitmap
428 if ( k < BUFR_MAX_QUALITY_DATA )
429 ( b->bitmap.bmap[b->bitmap.nba - 1]->ns1 )++;
430 else
431 {
432 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence(): No more space for first order statistic vars in bitmap. Check BUFR_MAX_QUALITY_DATA\n" );
433 return 1;
434 }
435 }
436 s->sequence[s->nd].related_to = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop];
437 }
438
439 //case of difference statistics
440 if ( b->state.dstat_active &&
441 l->lseq[i].x == 8 && l->lseq[i].y == 24 )
442 {
443 if ( ixloop == 0 )
444 {
445 k = b->bitmap.bmap[b->bitmap.nba - 1]->nds; // index in stats
446 b->bitmap.bmap[b->bitmap.nba - 1]->dstat_desc[k] = s->nd; // Set the value of statistical parameter
447 // update the number of quality variables for the bitmap
448 if ( k < BUFR_MAX_QUALITY_DATA )
449 ( b->bitmap.bmap[b->bitmap.nba - 1]->nds )++;
450 else
451 {
452 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence(): No more space for difference statistic vars in bitmap. Check BUFR_MAX_QUALITY_DATA\n" );
453 return 1;
454 }
455 }
456 s->sequence[s->nd].related_to = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop];
457 }
458
459 if ( s->nd < ( s->dim - 1 ) )
460 {
461 ( s->nd ) ++;
462 }
463 else if ( bufrdeco_increase_data_array ( s ) == 0 )
464 {
465 ( s->nd ) ++;
466 }
467 else
468 {
469 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence(): No more bufr_atom_data available. Check BUFR_NMAXSEQ\n" );
470 return 1;
471 }
472 break;
473
474 case 1:
475 // Case of replicator descriptor
476 replicator.s = l;
477 replicator.ixrep = i;
478 if ( l->lseq[i].y != 0 )
479 {
480 // no delayed
481 replicator.ixdel = i;
482 replicator.ndesc = l->lseq[i].x;
483 replicator.nloops = l->lseq[i].y;
484 bufrdeco_decode_replicated_subsequence ( s, &replicator, b );
485 ixd += replicator.ndesc; // update ixd properly
486 }
487 else
488 {
489 // case of delayed;
490 replicator.ixdel = i + 1;
491 replicator.ndesc = l->lseq[i].x;
492 // here we read ndesc from delayed replicator descriptor
493 if ( bufrdeco_tableb_val ( & ( s->sequence[s->nd] ), b, & ( l->lseq[i + 1] ) ) )
494 {
495 return 1;
496 }
497 replicator.nloops = ( size_t ) s->sequence[s->nd].val;
498 if ( s->nd < ( s->dim - 1 ) )
499 {
500 ( s->nd ) ++;
501 }
502 else if ( bufrdeco_increase_data_array ( s ) == 0 )
503 {
504 ( s->nd ) ++;
505 }
506 else
507 {
508 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence(): No more bufr_atom_data available. Check BUFR_NMAXSEQ\n" );
509 return 1;
510 }
511 bufrdeco_decode_replicated_subsequence ( s, &replicator, b );
512 ixd += replicator.ndesc + 1; // update ixd properly
513 }
514 //i = r->ixdel + r->ndesc; // update i properly
515 break;
516
517 case 2:
518 // Case of operator descriptor
519 if ( bufrdeco_parse_f2_descriptor ( s, & ( l->lseq[i] ), b ) )
520 {
521 return 1;
522 }
523
524 // Case of subsituted values
525 if ( b->state.subs_active && l->lseq[i].x == 23 && l->lseq[i].y == 255 )
526 {
527 k = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop]; // ref which is bitmaped_to
528 // Get the bitmaped descriptor k
529 if ( ixloop == 0 )
530 {
531 b->bitmap.bmap[b->bitmap.nba - 1]->subs = s->nd;
532 }
533 if (bufrdeco_tableb_val ( & ( s->sequence[s->nd] ), b, & ( l->lseq[k] ) ) )
534 {
535 return 1;
536 }
537 s->sequence[s->nd].related_to = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop];
538
539 if ( s->nd < ( s->dim - 1 ) )
540 {
541 s->nd += 1;
542 }
543 else
544 {
545 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence(): Reached limit. Consider increas BUFR_NMAXSEQ\n" );
546 return 1;
547 }
548 }
549
550 // Case of repaced/retained values
551 if ( b->state.retained_active && l->lseq[i].x == 32 && l->lseq[i].y == 255 )
552 {
553 k = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop];
554 // Get the bitmaped descriptor k
555 if ( ixloop == 0 )
556 {
557 b->bitmap.bmap[b->bitmap.nba - 1]->retain = s->nd;
558 }
559 if ( bufrdeco_tableb_val ( & ( s->sequence[s->nd] ), b, & ( l->lseq[k] ) ) )
560 {
561 return 1;
562 }
563
564 s->sequence[s->nd].related_to = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop];
565 if ( s->nd < ( s->dim - 1 ) )
566 {
567 s->nd += 1;
568 }
569 else
570 {
571 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence(): Reached limit. Consider increas BUFR_NMAXSEQ\n" );
572 return 1;
573 }
574 }
575
576 // Case of first order statistics
577 //
578 // This operator shall signify a data item containing a
579 // first-order statistical value of the type indicated by
580 // the preceding 0 08 023 element descriptor; the
581 // element descriptor to which the first-order statistic
582 // relates is obtained by the application of the data
583 // present bit-map associated with the first-order
584 // statistical values follow operator; first-order statistical
585 // values shall be represented as defined by this element
586 // descriptor.
587 if ( b->state.stat1_active && l->lseq[i].x == 24 && l->lseq[i].y == 255 )
588 {
589 k = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop];
590
591 // Get the bitmaped descriptor k
592 if ( ixloop == 0 )
593 {
594 b->bitmap.bmap[b->bitmap.nba - 1]->stat1[b->bitmap.bmap[b->bitmap.nba - 1]->ns1 -1] = s->nd;
595 }
596
597 if ( bufrdeco_tableb_val ( & ( s->sequence[s->nd] ), b, & ( l->lseq[k] ) ) )
598 {
599 return 1;
600 }
601
602 s->sequence[s->nd].related_to = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop];
603 if ( s->nd < ( s->dim - 1 ) )
604 {
605 s->nd += 1;
606 }
607 else
608 {
609 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence(): Reached limit. Consider increas BUFR_NMAXSEQ\n" );
610 return 1;
611 }
612 }
613
614 // Case of difference statistics
615 //
616 // This operator shall signify a data item containing a
617 // difference statistical value of the type indicated by
618 // the preceding 0 08 024 element descriptor; the
619 // element descriptor to which the difference statistical
620 // value relates is obtained by the application of the
621 // data present bit-map associated with the difference
622 // statistical values follow operator; difference statistical
623 // values shall be represented as defined by this
624 // element descriptor, but with a reference value of 2**n
625 // and a data width of (n+1), where n is the data width
626 // given by the original descriptor. This special reference
627 // value allows the statistical difference values to
628 // be centred around zero.
629 if ( b->state.dstat_active && l->lseq[i].x == 25 && l->lseq[i].y == 255 )
630 {
631 k = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop];
632
633 // Get the bitmaped descriptor k
634 if ( ixloop == 0 )
635 {
636 b->bitmap.bmap[b->bitmap.nba - 1]->stat1[b->bitmap.bmap[b->bitmap.nba - 1]->nds -1] = s->nd;
637 }
638
639 // in bufrdeco_tableb_val() is taken into acount when is difference statistics active
640 if ( bufrdeco_tableb_val ( & ( s->sequence[s->nd] ), b, & ( l->lseq[k] ) ) )
641 {
642 return 1;
643 }
644
645 /*s->sequence[s->nd].ref = - ( ( int32_t ) 1 << s->sequence[s->nd].bits );
646 s->sequence[s->nd].bits++;*/
647 s->sequence[s->nd].related_to = b->bitmap.bmap[b->bitmap.nba - 1]->bitmap_to[ixloop];
648 if ( s->nd < ( s->dim - 1 ) )
649 {
650 s->nd += 1;
651 }
652 else
653 {
654 sprintf ( b->error, "bufrdeco_decode_replicated_subsequence(): Reached limit. Consider increas BUFR_NMAXSEQ\n" );

662 // Case of sequence descriptor
663 if ( bufrdeco_decode_subset_data_recursive ( s, l->sons[i], b ) )
