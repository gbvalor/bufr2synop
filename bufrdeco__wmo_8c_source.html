153/*!
154 \fn int bufr_read_tables (struct bufrdeco *b)
155 \brief Read the tables according with bufr file data from a bufr table directory
156 \param b basic struct with needed data
157
158 The default directories where to search bufr tables are stored in \ref DEFAULT_BUFRTABLES_WMO_CSV_DIR1 and \ref DEFAULT_BUFRTABLES_WMO_CSV_DIR2
159*/
160int bufr_read_tables ( struct bufrdeco *b )
161{
162 int index;
163 buf_t i;
164 bufrdeco_assert ( b != NULL );
165 struct bufr_tableB *tb;
166
167 if ( b->mask & BUFRDECO_USE_TABLES_CACHE )
168 {
169 // When using cache, member b->tables is actually a pointer in array b->cache.tab[]
170
171 if ( ( index = bufrdeco_cache_tables_search ( & ( b->cache ), b->sec1.master_version ) ) >= 0 )
172 {
173#ifdef __DEBUG
174 printf ( "# Found tables in cache for version %u index %d\n", b->sec1.master_version, index );
175#endif
176 // hit cache, then the only task is to change member b->tables, and restore original values from item im tableB
177 b->tables = b->cache.tab[index];
178 tb = & ( b->tables->b );
179
180 for ( i = 0; i < tb->nlines ; i++ )
181 {
182 tb->item[i].scale = tb->item[i].scale_ori;
183 tb->item[i].reference = tb->item[i].reference_ori;
184 tb->item[i].nbits = tb->item[i].nbits_ori;
185 tb->item[i].changed = 0;
186 }
187
188 // all done
189 return 0;
190 }
191 else
192 {
193#ifdef __DEBUG
194 printf ( "# Tables for version %u not found in cache. Stored in index %d\n", b->sec1.master_version, b->cache.next );
195#endif
196 // If not in cache, the new master version tables has to be stored. This implies that
197 bufrdeco_store_tables ( & ( b->tables ), & ( b->cache ), b->sec1.master_version );
198
199 // get tablenames
200 if ( get_wmo_tablenames ( b ) )
201 {
202 snprintf ( b->error, sizeof ( b->error ),"%s(): Cannot find bufr tables\n", __func__ );
203 return 1;
204 }
205
206 // Missed cache
207 if ( bufr_read_tableB ( b ) )
208 {
209 return 1;
210 }
211
212 if ( bufr_read_tableC ( b ) )
213 {
214 return 1;
215 }
216
217 if ( bufr_read_tableD ( b ) )
218 {
219 return 1;
220 }
221
222 }
223 }
224 else
225 {
226 // If tables still not initialized then do it
227 if ( b->tables == NULL && bufrdeco_init_tables ( & ( b->tables ) ) )
228 {
229 snprintf ( b->error, sizeof ( b->error ), "%s(): Cannot allocate memory for tables\n", __func__ );
230 return 1;
231 }
232 // get tablenames
233 if ( get_wmo_tablenames ( b ) )
234 {
235 snprintf ( b->error, sizeof ( b->error ),"%s(): Cannot find bufr tables\n", __func__ );
236 return 1;
237 }
238
239 // And now read tables
240 if ( bufr_read_tableB ( b ) )
241 {
242 return 1;
243 }
244
245 if ( bufr_read_tableC ( b ) )
246 {
247 return 1;
248 }
249
250 if ( bufr_read_tableD ( b ) )
251 {
252 return 1;
253 }
254 }
255 return 0;
256}

258/*!
259 * \fn int bufrdeco_store_tables ( struct bufr_tables **t, struct bufr_tables_cache *c, uint8_t ver )
260 * \brief Init an element of array c->tab[] if still not allocated. If allocated clean it and set *t pointing to this element
261 *
262 */
263int bufrdeco_store_tables ( struct bufr_tables **t, struct bufr_tables_cache *c, uint8_t ver )
264{
265 if ( c->tab[c->next] == NULL )
266 {
267 // Init the array element
268 bufrdeco_init_tables ( & ( c->tab[c->next] ) );
269
270 // increase the counter of allocated elements
271 ( c->nt )++;
272 }
273 else
274 {
275 // Clean the element in array with zeroes
276 memset ( c->tab[c->next], 0, sizeof ( struct bufr_tables ) );
277
278 // sets the proper version as a key of element
279 c->ver[c->next] = ver;
280 }
281
282 // t will point to array element
283 *t = c->tab[c->next];
284
285 // Set the member 'next'. It marks the next element in array it will be modified when failing a search in future
286 c->next = ( c->next + 1 ) % BUFRDECO_TABLES_CACHE_SIZE;
287 return 0;
288}

290/*!
291 * \fn int bufrdeco_cache_tables_search ( struct bufr_tables_cache *c, uint8_t ver )
292 * \brief Search a struct \ref bufr_tables in \ref bufr_tables_cache
293 * \param c pointer to the struct \ref bufr_tables_cache where to search
294 * \param ver small int with the master version acting as a key in the seardch
295 *
296 * Returns the index of found struct. If no struct found returns -1
297 */
298int bufrdeco_cache_tables_search ( struct bufr_tables_cache *c, uint8_t ver )
299{
300 buf_t i = 0;
301
302 for ( i = 0; i < BUFRDECO_TABLES_CACHE_SIZE ; i++ )
303 {
304 if ( c->ver[i] == ver )
305 return i; // found
306 }
307 return -1; // Not found
308}

310/*!
311 * \fn int bufrdeco_free_cache_tables (struct bufr_tables_cache *c)
312 * \brief deallocate and clean a \ref bufr_tables_cache
313 * \param c pointer to the struct to clean
314 */
315int bufrdeco_free_cache_tables ( struct bufr_tables_cache *c )
316{
317 buf_t i;
318
319 for ( i = 0; i < BUFRDECO_TABLES_CACHE_SIZE; i++ )
320 {
321 if ( c->tab[i] )
322 {
323 free ( ( void * ) c->tab[i] );
324 }
325 }
326 // then clean
327 memset ( c, 0, sizeof ( struct bufr_tables_cache ) );
328 return 0;
329}
