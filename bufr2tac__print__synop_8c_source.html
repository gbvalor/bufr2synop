32/*!
33 \fn char * print_synop_sec0 (char **sec0, size_t lmax, struct synop_chunks *syn)
34 \brief Prints the synop section 0 (header)
35 \param sec0 the pointer where to print section
36 \param lmax max length permited
37 \param syn pointer to s atruct \ref synop_chunks where the parse results are set
38*/
39char * print_synop_sec0 ( char **sec0, size_t lmax, struct synop_chunks *syn )
40{
41 char *c = *sec0;
42
43 if ( check_len ( sec0,12 ) )
44 {
45 c += sprintf ( c, "%s%s%s%s%s", syn->e.YYYY, syn->e.MM, syn->e.DD, syn->e.HH, syn->e.mm );
46 }
47
48 // Print type
49 if ( check_len ( sec0,5 ) )
50 {
51 c += sprintf ( c, " %s%s", syn->s0.MiMi, syn->s0.MjMj );
52 }
53
54 if ( check_len ( sec0,6 ) )
55 {
56 if ( syn->s0.D_D[0] )
57 {
58 c += sprintf ( c, " %s", syn->s0.D_D );
59 }
60 else if ( syn->s0.A1[0] && syn->s0.bw[0] && syn->s0.nbnbnb[0] )
61 {
62 c += sprintf ( c, " %s%s%s", syn->s0.A1, syn->s0.bw, syn->s0.nbnbnb );
63 }
64
65 }
66
67 // print YYGGiw
68 if ( check_len ( sec0,6 ) )
69 {
70 c += sprintf ( c, " %s%s%s", syn->s0.YY, syn->s0.GG, syn->s0.iw );
71 }
72
73 // print IIiii
74 if ( check_len ( sec0,6 ) && syn->s0.II[0] )
75 {
76 c += sprintf ( c, " %s%s", syn->s0.II, syn->s0.iii );
77 }
78 else
79 {
80 if ( syn->s0.LaLaLa[0] )
81 {
82 c += sprintf ( c, " 99%s", syn->s0.LaLaLa );
83 }
84 else
85 {
86 c += sprintf ( c, " 99///" );
87 }
88
89 if ( syn->s0.Qc[0] && syn->s0.LoLoLoLo[0] )
90 {
91 c += sprintf ( c, " %s%s", syn->s0.Qc, syn->s0.LoLoLoLo );
92 }
93 else
94 {
95 c += sprintf ( c, " /////" );
96 }
97 }
98
99 if ( strcmp ( syn->s0.MiMi, "OO" ) == 0 )
100 {
101 if ( syn->s0.MMM[0] && syn->s0.Ula[0] && syn->s0.Ulo[0] )
102 {
103 c += sprintf ( c, " %s%s%s", syn->s0.MMM, syn->s0.Ula, syn->s0.Ulo );
104 }
105 if ( syn->s0.h0h0h0h0[0] )
106 {
107 c += sprintf ( c, " %s%s", syn->s0.h0h0h0h0, syn->s0.im );
108 }
109
110 }
111
112 *sec0 = c;
113 return *sec0;
114}
