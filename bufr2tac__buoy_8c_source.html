26 /*!
27  \fn int buoy_YYYYMMDDHHmm_to_JMMYYGGgg(struct buoy_chunks *b)
28  \brief Sets YYGG from YYYYMMDDHHmm extended group
29  \param b pointer to the target struct \ref buoy_chunks
30 */
31 int buoy_YYYYMMDDHHmm_to_JMMYYGGgg ( struct buoy_chunks *b )
32 {
33  char aux[20];
34  time_t t;
35  struct tm tim;
36 
37  if ( strlen ( b->e.YYYY ) &&
38  strlen ( b->e.MM ) &&
39  strlen ( b->e.DD ) &&
40  strlen ( b->e.HH ) &&
41  strlen ( b->e.MM ) )
42  {
43  sprintf ( aux,"%s%s%s%s%s", b->e.YYYY, b->e.MM, b->e.DD, b->e.HH, b->e.mm );
44  }
45 
46  if ( strlen ( aux ) != 12 )
47  {
48  return 1;
49  }
50 
51  memset ( &tim, 0, sizeof ( struct tm ) );
52  strptime ( aux, "%Y%m%d%H%M", &tim );
53 
54  t = mktime ( &tim );
55  gmtime_r ( &t, &tim );
56  sprintf ( b->s0.YY, "%02d", tim.tm_mday );
57  sprintf ( b->s0.GG, "%02d", tim.tm_hour );
58  sprintf ( b->s0.MM, "%02d", tim.tm_mon + 1 );
59  sprintf ( b->s0.J, "%d", tim.tm_year % 10 );
60  sprintf ( b->s0.gg, "%02d", tim.tm_min );
61 
62  return 0;
63 }

67  \brief parses a subset sequence as an Buoy SYNOP FM-18 report
68  \param m pointer to a struct \ref metreport where set some results
69  \param sq pointer to a struct \ref bufr_subset_sequence_data with the parsed sequence on input
70  \param err string with detected errors, if any
71 
72  It return 0 if all is OK. Otherwise it also fills the \a err string
73 */
74 int parse_subset_as_buoy ( struct metreport *m, struct bufr2tac_subset_state *s, struct bufr_subset_sequence_data *sq, char *err )
75 {
76  size_t is;
77  char aux[16];
78  struct buoy_chunks *b;
79 
80  //
81  b = &m->buoy;
82 
83  // clean data
84  clean_buoy_chunks ( b );
85 
86  // reject if still not coded type
87  if ( strcmp ( s->type_report,"ZZYY" ) )
88  {
89  sprintf ( err,"bufr2tac: '%s' report still not decoded in this software", s->type_report );
90  return 1;
91  }
92 
93  strcpy ( b->s0.MiMi, "ZZ" );
94  strcpy ( b->s0.MjMj, "YY" );
95 
96  b->mask = BUOY_SEC0;
97 
98  /**** First pass, sequential analysis *****/
99  for ( is = 0; is < sq->nd; is++ )
100  {
101  // check if is a significance qualifier
102  if ( sq->sequence[is].desc.x == 8 )
103  {
104  s->i = is;
105  s->a = &sq->sequence[is];
106  buoy_parse_x08 ( b, s );
107  }
108 
109  if ( sq->sequence[is].mask & DESCRIPTOR_VALUE_MISSING ||
110  s->isq // case of an significance qualifier
111  )
112  {
113  continue;
114  }
115 
116  s->i = is;
117  s->ival = ( int ) sq->sequence[is].val;
118  s->val = sq->sequence[is].val;
119  s->a = &sq->sequence[is];
120  if ( is > 0 )
121  {
122  s->a1 = &sq->sequence[is - 1];
123  }
124 
125  switch ( sq->sequence[is].desc.x )
126  {
127 
128  case 1: //localization descriptors
129  buoy_parse_x01 ( b, s );
130  break;
131 
132  case 2: //Date time descriptors
133  buoy_parse_x02 ( b, s );
134  break;
135 
136  case 4: //Date time descriptors
137  buoy_parse_x04 ( b, s );
138  break;
139 
140  case 5: //Position
141  buoy_parse_x05 ( b, s );
142  break;
143 
144  case 6: // Horizontal Position -2
145  buoy_parse_x06 ( b, s );
146  break;
147 
148  case 7: // Vertical Position
149  buoy_parse_x07 ( b, s );
150  break;
151 
152  case 10: // Air Pressure descriptors
153  buoy_parse_x10 ( b, s );
154  break;
155 
156  case 11: // wind data
157  buoy_parse_x11 ( b, s );
158  break;
159 
160  case 12: //Temperature descriptors
161  buoy_parse_x12 ( b, s );
162  break;
163 
164  case 13: // Humidity and precipitation data
165  buoy_parse_x13 ( b, s );
166  break;
167 
168  case 14: // Radiation
169  buoy_parse_x14 ( b, s );
170  break;
171 
172  case 20: // Cloud data
173  buoy_parse_x20 ( b, s );
174  break;
175 
176  case 22: // Oceanographic data
177  buoy_parse_x22 ( b, s );
178  break;
179 
180  case 31: // Replicators
181  buoy_parse_x31 ( b, s );
182  break;
183 
184 
185  case 33: // Quality data
186  buoy_parse_x33 ( b, s );
187  break;
188 
189 
190  default:
191  break;
192  }
193 
194  }
195 
196  if ( ( ( s->mask & SUBSET_MASK_HAVE_LATITUDE ) == 0 ) ||
197  ( ( s->mask & SUBSET_MASK_HAVE_LONGITUDE ) == 0 ) ||
198  ( ( s->mask & SUBSET_MASK_HAVE_YEAR ) == 0 ) ||
199  ( ( s->mask & SUBSET_MASK_HAVE_MONTH ) == 0 ) ||
200  ( ( s->mask & SUBSET_MASK_HAVE_DAY ) == 0 ) ||
201  ( ( s->mask & SUBSET_MASK_HAVE_HOUR ) == 0 ) )
202  {
203  sprintf ( err,"bufr2tac: parse_subset_as_buoy(): lack of mandatory descriptor in sequence" );
204  return 1;
205  }
206 
207 
208  /****** Second pass. Global results and consistence analysis ************/
209 
210  // adjust iw
211  if ( b->s0.iw[0] == '/' && b->s1.ff[0] != '/' )
212  {
213  b->s0.iw[0] = '1';
214  }
215 
216  // fill date fields YYYYMMDDHHmm
217  buoy_YYYYMMDDHHmm_to_JMMYYGGgg ( b );
218  b->mask |= BUOY_EXT;
219 
220  // Fill some metreport fields
221  if ( s->mask & SUBSET_MASK_HAVE_LATITUDE )
222  {
223  if ( fabs ( s->lat ) <= 90.0 )
224  {
225  m->g.lat = s->lat;
226  }
227  else
228  {
229  return 1; // Bad latitude. Fatal error
230  }
231  }
232  if ( s->mask & SUBSET_MASK_HAVE_LONGITUDE )
233  {
234  if ( fabs ( s->lon ) <= 180.0 )
235  {
236  m->g.lon = s->lon;
237  }
238  else
239  {
240  return 1; // bad longitude. Fatal error
241  }
242  }
243  if ( s->mask & SUBSET_MASK_HAVE_ALTITUDE )
244  {
245  m->g.alt = s->alt;
246  }
247  if ( s->mask & SUBSET_MASK_HAVE_NAME )
248  {
249  strcpy ( m->g.name, s->name );
250  }
251  sprintf ( aux,"%s%s%s%s%s", b->e.YYYY, b->e.MM, b->e.DD, b->e.HH, b->e.mm );
252  YYYYMMDDHHmm_to_met_datetime ( &m->t, aux );
253 
254  // Fill some metreport fields
255  if ( b->s0.A1[0] && b->s0.bw[0] && b->s0.nbnbnb[0] )
256  {
257  sprintf ( m->g.index, "%s%s%s", b->s0.A1, b->s0.bw, b->s0.nbnbnb );
258  }
259 
260  // check if set both LaLaLa and LoLoLoLo to set Qc
261  if ( ( b->s0.Qc[0] == 0 ) && b->s0.LaLaLaLaLa[0] && b->s0.LoLoLoLoLoLo[0] )
262  {
263  if ( s->mask & SUBSET_MASK_LATITUDE_SOUTH )
264  {
265  if ( s->mask & SUBSET_MASK_LONGITUDE_WEST )
266  {
267  strcpy ( b->s0.Qc, "5" );
268  }
269  else
270  {
271  strcpy ( b->s0.Qc, "3" );
272  }
273  }
274  else
275  {
276  if ( s->mask & SUBSET_MASK_LONGITUDE_WEST )
277  {
278  strcpy ( b->s0.Qc, "7" );
279  }
280  else
281  {
282  strcpy ( b->s0.Qc, "1" );
283  }
284  }
285  }
286 
287  // check Qx
288  if ( b->s2.Qd[0] == 0 )
289  {
290  b->s2.Qd[0] = '0';
291  }
292  if ( b->s3.k2[0] == 0 )
293  {
294  b->s3.k2[0] = '0';
295  }
296  if ( b->s1.Qx[0] == 0 && b->s1.Qd[0] )
297  {
298  b->s1.Qx[0] = '9';
299  }
300  if ( b->s2.Qx[0] == 0 && b->s2.Qd[0] )
301  {
302  b->s2.Qx[0] = '9';
303  }
304 
305  if ( b->s3.Qd1[0] == 0 )
306  {
307  b->s3.Qd1[0] = '0';
308  }
309  if ( b->s3.Qd2[0] == 0 )
310  {
311  b->s3.Qd2[0] = '0';
312  }
313 
314  if ( b->s3.k3[0] == 0 )
315  {
316  b->s3.k3[0] = '/';
317  }
318  if ( b->s3.k6[0] == 0 )
319  {
320  b->s3.k6[0] = '/';
321  }
322 
323  return 0;
324 }
