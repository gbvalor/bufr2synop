26 /*!
27  \fn char * pascal_to_ppp ( char *target, double P )
28  \brief Converts pascal values (variation) into a ppp string
29  \param P the pressure variation in Pascal units
30  \param target string with the result
31 */
32 char * pascal_to_ppp ( char *target, double P )
33 {
34  int ic;
35  if ( P > 0 )
36  {
37  ic = ( int ) ( P * 0.1 );
38  }
39  else
40  {
41  ic = ( int ) ( -P * 0.1 );
42  }
43  sprintf ( target, "%03d", ic % 1000 );
44  return target;
45 }

77 /*!
78  \fn int syn_parse_x10 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
79  \brief Parse a expanded descriptor with X = 10
80  \param syn pointer to a struct \ref synop_chunks where to set the results
81  \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
82 
83  It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
84 */
85 int syn_parse_x10 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
86 {
87  char aux[16];
88 
89  if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
90  {
91  return 0;
92  }
93 
94  switch ( s->a->desc.y )
95  {
96  case 4: // 0 10 004 . Pressure
97  pascal_to_PPPP ( aux, s->val );
98  strcpy ( syn->s1.PoPoPoPo, aux );
99  syn->mask |= SYNOP_SEC1;
100  break;
101 
102  case 9: // 0 10 009. Geopotential at standard level
103  sprintf ( syn->s1.hhh, "%03d", s->ival % 1000 );
104  syn->mask |= SYNOP_SEC1;
105  break;
106 
107  case 51: // 0 10 051 . Pressure reduced to mean sea level
108  pascal_to_PPPP ( aux, s->val );
109  strcpy ( syn->s1.PPPP, aux );
110  syn->mask |= SYNOP_SEC1;
111  break;
112 
113  case 61: // 0 10 061 . 3-hour pressure change
114  pascal_to_ppp ( aux, s->val );
115  sprintf ( syn->s1.ppp, "%s", aux );
116  syn->mask |= SYNOP_SEC1;
117  break;
118 
119  case 62: // 0 10 062 . 24-hour pressure change
120  pascal_to_ppp ( aux, s->val );
121  sprintf ( syn->s3.ppp24, "%s", aux );
122  if ( s->val >= 0 )
123  {
124  strcpy ( syn->s3.snp24, "8" );
125  }
126  else
127  {
128  strcpy ( syn->s3.snp24, "9" );
129  }
130  syn->mask |= SYNOP_SEC3;
131  break;
132 
133  case 63: // 0 10 063 . Characteristic of pressure tendency
134  sprintf ( syn->s1.a, "%1d",s->ival );
135  syn->mask |= SYNOP_SEC1;
136  break;
137  default:
138  break;
139  }
140  return 0;
141 }
