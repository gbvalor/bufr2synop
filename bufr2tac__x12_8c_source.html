26/*!
27 \fn char * kelvin_to_snTTT ( char *target, double T )
28 \brief converts a kelvin temperature value into a snTTT string
29 \param T the temperature ( Kelvin )
30 \param target string with the result
31*/
32char * kelvin_to_snTTT ( char *target, double T )
33{
34 int ic;
35 double Tx = T;
36
37 if ( Tx > 15 && Tx < 34.0 ) // guess there is an scale error on coding , we better manage T * 10
38 {
39 Tx *= 10.0;
40 }
41 else if ( T < 173.205 || T > 340.0 )
42 {
43 return NULL;
44 }
45
46 // tenths of degree (Celsius)
47 // Patch introduced on 22-Dec-2019 due to some roundings errors from India and Australia
48 // not converting properly from Kelvin to Celsius for Tmax and Tmin
49 // Changed '0.5' to '0.45'
50 //ic = ( int ) ( floor ( 10.0 * ( Tx - 273.15 ) + 0.5 ) );
51 ic = ( int ) ( floor ( 10.0 * ( Tx - 273.15 ) + 0.45 ) );
52 if ( ic < 0 )
53 {
54 sprintf ( target, "1%03d", -ic );
55 }
56 else
57 {
58 sprintf ( target, "0%03d", ic );
59 }
60 return target;
61}

66 \param T the temperature ( Kelvin )

69char * kelvin_to_snTT ( char *target, double T )

71 int ic;
72 if ( T < 173.65 || T > 340.0 )
73 {
74 return NULL;
75 }
76
77 // Whole degrees (Celsius)
78 ic = ( int ) ( floor ( ( T - 273.15 ) + 0.5 ) );
79 if ( ic < 0 )
80 {
81 sprintf ( target, "1%02d", -ic );
82 }
83 else
84 {
85 sprintf ( target, "%03d", ic );

90/*!
91 \fn char * kelvin_to_TT ( char *target, double T )
92 \brief converts a kelvin temperature value into a TT string
93 \param T the temperature ( Kelvin )
94 \param target string with the result
95*/
96char * kelvin_to_TT ( char *target, double T )
97{
98 int ic;
99 if ( T < 150.0 || T > 340.0 )
100 {
101 return NULL;
102 }
103
104 // Whole degrees (Celsius)
105 ic = ( int ) ( floor ( ( T - 273.15 ) + 0.5 ) );
106 if ( ic < 0 )
107 {
108 sprintf ( target, "%02d", 50 - ic );
109 }
110 else
111 {
112 sprintf ( target, "%02d", ic );
113 }
114 return target;

117/*!
118 \fn char * kelvin_to_TTTT ( char *target, double T )
119 \brief converts a kelvin temperature value into a TTTT string
120 \param T the temperature ( Kelvin )
121 \param target string with the result
122*/
123char * kelvin_to_TTTT ( char *target, double T )
124{
125 int ic;
126 if ( T < 150.0 || T > 340.0 )
127 {
128 return NULL;
129 }
130 // hundreth of degrees (Celsius)
131 ic = ( int ) ( floor ( 100.0 * ( T - 273.15 ) + 0.5 ) );
132 if ( ic < 0 )
133 {
134 ic = 5000 - ic;
135 }
136 sprintf ( target, "%04d", ic );
137 return target;
138}

141/*!
142 \fn char * kelvin_to_TTTa(char *target, double T)
143 \brief Set temperature TTTa
144 \param target result as string
145 \param T temperature (kelvin)
146*/
147char * kelvin_to_TTTa ( char *target, double T )
148{
149 int ix;
150
151 if ( T == MISSING_REAL )
152 {
153 sprintf ( target, "///" );
154 return target;
155 }
156
157 ix = ( int ) ( 10.0 * ( T - 273.15 ) );
158
159 if ( ix >= 0 )
160 {
161 ix &= ~1;
162 }
163 else
164 {
165 ix = ( -ix );
166 ix |= 1;
167 }
168 sprintf ( target, "%03d", ix );
169 return target;
170}

172/*!
173 \fn char * dewpoint_depression_to_DnDn ( char * target, double T, double Td )
174 \brief Set DnDn (dewpoint depression)
175 \param target string to set as result
176 \param T temperature (Kelvin)
177 \param Td dewpoint (Kelvin)
178*/
179char * dewpoint_depression_to_DnDn ( char * target, double T, double Td )
180{
181 int ix;
182 double dpd = T - Td;
183
184 if ( T == MISSING_REAL || Td == MISSING_REAL )
185 {
186 strcpy ( target, "//" );
187 return target;
188 }
189
190 if ( dpd < 0.0 )
191 {
192 return NULL;
193 }
194
195 if ( dpd > 50.0 )
196 {
197 strcpy ( target, "//" );
198 return target;
199 }
200
201 ix = ( int ) ( 10.0 * dpd );
202
203 if ( ix >= 55 )
204 {
205 ix = 50 + ( ix + 5 ) / 10;
206 }
207 sprintf ( target, "%02d", ix );
208
209 return target;
210}

212/*!
213 \fn int syn_parse_x12 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
214 \brief Parse a expanded descriptor with X = 12
215 \param syn pointer to a struct \ref synop_chunks where to set the results
216 \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
217
218 It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
219*/
220int syn_parse_x12 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
221{
222 char aux[16];
223
224 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
225 {
226 return 0;
227 }
228
229 switch ( s->a->desc.y )
230 {
231 case 1: // 0 12 001 . Air temperature
232 case 4: // 0 12 004 . Air temperatura at 2 m
233 case 101: // 0 12 101 . Air temperature
234 case 104: // 0 12 104 . Air temperature at 2 m
235 if ( syn->s1.TTT[0] == 0 )
236 {
237 if ( kelvin_to_snTTT ( aux, s->val ) )
238 {
239 syn->s1.sn1[0] = aux[0];
240 strcpy ( syn->s1.TTT, aux + 1 );
241 syn->mask |= SYNOP_SEC1;
242 }
243 }
244 break;
245
246 case 3: // 0 12 003 . Dewpoint temperature
247 case 6: // 0 12 006 . Dewpoint temperature at 2 m
248 case 103: // 0 12 103 . Dewpoint temperature
249 case 106: // 0 12 106 . Dewpoint temperature at 2 m
250 if ( syn->s1.TdTdTd[0] == 0 )
251 {
252 if ( kelvin_to_snTTT ( aux, s->val ) )
253 {
254 syn->s1.sn2[0] = aux[0];
255 strcpy ( syn->s1.TdTdTd, aux + 1 );
256 syn->mask |= SYNOP_SEC1;
257 }
258 }
259 break;
260
261 case 14: // 0 12 014 . Maximum temperature at 2 m , past 12 hours
262 case 21: // 0 12 021 . Maximum temperature.
263 case 114: // 0 12 114 . Maximum temperature at 2 m , past 12 hours
264 case 116: // 0 12 116 . Maximum temperature at 2 m , past 24 hours
265 if ( syn->s3.TxTxTx[0] == 0 && ( s->itval % ( 3 * 3600 ) ) == 0 ) // only for 3, 6 ... hours
266 {
267 if ( kelvin_to_snTTT ( aux, s->val ) )
268 {
269 syn->s3.snx[0] = aux[0];
270 strcpy ( syn->s3.TxTxTx, aux + 1 );
271 syn->mask |= SYNOP_SEC3;
272 }
273 }
274 break;
275
276 case 11: // 0 12 011 . Maximum temperature at heigh and over the period specified
277 case 111: // 0 12 111 . Maximum temperature at heigh and over the period specified
278 if ( syn->s3.TxTxTx[0] == 0 &&
279 ( s->hsensor >= 1.0 || s->hsensor <= 0 ) &&
280 ( time_period_duration ( s ) % ( 12 * 3600 ) ) == 0 ) // only for 12 , 24 hours
281 {
282 if ( kelvin_to_snTTT ( aux, s->val ) )
283 {
284 syn->s3.snx[0] = aux[0];
285 strcpy ( syn->s3.TxTxTx, aux + 1 );
286 syn->mask |= SYNOP_SEC3;
287 }
288 }
289 break;
290
291
292 case 15: // 0 12 015 . Minimum temperature at 2 m , past 12 hours
293 case 22: // 0 12 022 . Minimum temperature.
294 case 115: // 0 12 115 . Minimum temperature at 2 m , past 12 hours
295 case 117: // 0 12 117 . Minimum temperature at 2 m , past 24 hours
296 if ( syn->s3.TnTnTn[0] == 0 && ( s->itval % ( 3 * 3600 ) ) == 0 ) // only for 3, 6 ... hours
297 {
298 if ( kelvin_to_snTTT ( aux, s->val ) )
299 {
300 syn->s3.snn[0] = aux[0];
301 strcpy ( syn->s3.TnTnTn, aux + 1 );
302 syn->mask |= SYNOP_SEC3;
303 }
304 }
305 break;
306
307 case 12: // 0 12 012 . Minimum temperature at heigh and over the period specified
308 case 112: // 0 12 112 . Minimum temperature at heigh and over the period specified
309 if ( syn->s3.TnTnTn[0] == 0 &&
310 ( s->hsensor >= 1.0 || s->hsensor <= 0 ) &&
311 ( time_period_duration ( s ) % ( 12 * 3600 ) ) == 0 ) // only for 12 , 24 hours
312 {
313 if ( kelvin_to_snTTT ( aux, s->val ) )
314 {
315 syn->s3.snn[0] = aux[0];
316 strcpy ( syn->s3.TnTnTn, aux + 1 );
317 syn->mask |= SYNOP_SEC3;
318 }
319 }
320 break;
321
322 case 2: // 0 12 002 . Wet bulb temperature
323 case 5: // 0 12 005 . Wet bulb temperature at 2 m
324 case 102: // 0 12 102 . Wet bulb temperature
325 case 105: // 0 12 105 . Wet bulb temperature at 2 m
326 if ( syn->s2.TbTbTb[0] == 0 )
327 {
328 if ( kelvin_to_snTTT ( aux, s->val ) )
329 {
330 syn->s2.sw[0] = aux[0];
331 strcpy ( syn->s2.TbTbTb, aux + 1 );
332 syn->mask |= SYNOP_SEC2;
333 }
334 }
335 break;
336
337 case 113: // 0 12 113 . Ground minimum temperature, past 12 hours
338 if ( strcmp ( "6", syn->s0.A1 ) == 0 ||
339 strcmp ( "3", syn->s0.A1 ) == 0 ||
340 strcmp ( "2", syn->s0.A1 ) == 0
341 ) // region III or VI
342 {
343 if ( kelvin_to_snTT ( aux, s->val ) )
344 {
345 strcpy ( syn->s3.jjj, aux );
346 syn->mask |= SYNOP_SEC3;
347 }
348 }
349 else if ( strcmp ( "1", syn->s0.A1 ) == 0 )
350 {
351 if ( kelvin_to_TT ( aux, s->val ) )
352 {
353 syn->s3.XoXoXoXo[0] = aux[0];
354 syn->s3.XoXoXoXo[1] = aux[1];
355 if ( syn->s3.XoXoXoXo[2] == 0 )
356 {
357 syn->s3.XoXoXoXo[2] = '/';
358 }
359 if ( syn->s3.XoXoXoXo[3] == 0 )
360 {
361 syn->s3.XoXoXoXo[3] = '/';
362 }
363 syn->mask |= SYNOP_SEC3;
364 }
365 }
366 break;
367
368 case 121: // 0 12 121 . Ground minimum temperature (at the time of observation)
369 if ( strcmp ( "2", syn->s0.A1 ) == 0 ) // region II
370 {
371 if ( kelvin_to_snTT ( aux, s->val ) )
372 {
373 syn->s3.XoXoXoXo[1] = aux[0];
374 syn->s3.XoXoXoXo[2] = aux[1];
375 syn->s3.XoXoXoXo[3] = aux[2];
376 syn->mask |= SYNOP_SEC3;
377 }
378 }
379 break;
380
381 case 120: // 0 12 120 . Ground temperature
382 if ( strcmp ( "2", syn->s0.A1 ) == 0 ) // Only for Region II
383 {
384 if ( kelvin_to_snTT ( aux, s->val ) )
385 {
386 if ( syn->s3.E1[0] )
387 syn->s3.XoXoXoXo[0] = '/';
388 else if ( syn->s3.E[0] )
389 syn->s3.XoXoXoXo[0] = syn->s3.E[0];
390 syn->s3.XoXoXoXo[1] = aux[0];
391 syn->s3.XoXoXoXo[2] = aux[1];
392 syn->s3.XoXoXoXo[3] = aux[2];
393 }
394 }
395 break;
396
397 case 122: // 0 12 122 . Ground minimum temperature during preceding night
398 if ( strcmp ( "1", syn->s0.A1 ) == 0 ) // Only for Region I
399 {
400 if ( kelvin_to_TT ( aux, s->val ) )
401 {
402 syn->s3.XoXoXoXo[0] = aux[0];
403 syn->s3.XoXoXoXo[1] = aux[1];
404 if ( syn->s3.XoXoXoXo[2] == 0 )
405 {
406 syn->s3.XoXoXoXo[2] = '/';
407 }
408 if ( syn->s3.XoXoXoXo[3] == 0 )
409 {
410 syn->s3.XoXoXoXo[3] = '/';
411 }
412 syn->mask |= SYNOP_SEC3;
413 }
414 }
415 else if ( strcmp ( "2", syn->s0.A1 ) == 0 ||
416 strcmp ( "3", syn->s0.A1 ) == 0 ) // Region II and III
417 {
418 if ( kelvin_to_snTT ( aux, s->val ) )
419 {
420 strcpy ( syn->s3.jjj, aux );
421 syn->mask |= SYNOP_SEC3;
422 }
423 }
424 break;
425
426 default:
427 break;
428 }
429 return 0;
430}

486/*!
487 \fn int climat_parse_x12 ( struct climat_chunks *c, struct bufr2tac_subset_state *s )
488 \brief Parse a expanded descriptor with X = 12
489 \param c pointer to a struct \ref climat_chunks where to set the results

491
492 It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
493*/
494int climat_parse_x12 ( struct climat_chunks *c, struct bufr2tac_subset_state *s )
495{
496 char aux[16];
497
498 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
499 {
500 return 0;
501 }
502
503 if ( c == NULL )
504 {
505 return 1;
506 }
507
508 switch ( s->a->desc.y )
509 {
510 case 101:
511 if ( s->is_normal == 0 )
512 {
513 if ( s->isq_val == 4 ) // mean value
514 {
515 if ( kelvin_to_snTTT ( aux, s->val ) )
516 {
517 c->s1.s[0] = aux[0];
518 strcpy ( c->s1.TTT, aux + 1 );
519 c->mask |= CLIMAT_SEC1;
520 }
521 }
522 else if ( s->isq_val == 2 ) // maximum value
523 {
524 if ( kelvin_to_snTTT ( aux, s->val ) )
525 {
526 c->s4.sax[0] = aux[0];
527 strcpy ( c->s4.Tax, aux + 1 );
528 if ( s->more_days == 0 )
529 {
530 sprintf ( c->s4.yax, "%02d", s->day );
531 }
532 else
533 {
534 sprintf ( c->s4.yax, "%02d", (s->day + 50) % 100 );
535 }
536 c->mask |= CLIMAT_SEC4;
537 }
538
539 }
540 else if ( s->isq_val == 3 ) // minimum value
541 {
542 if ( kelvin_to_snTTT ( aux, s->val ) )
543 {
544 c->s4.san[0] = aux[0];

557 }
558 else
559 {
560 if ( c->s2.TTT[0] == 0 )
561 {
562 if ( kelvin_to_snTTT ( aux, s->val ) )
563 {
564 c->s2.s[0] = aux[0];

566 c->mask |= CLIMAT_SEC2;

577 {
578 if ( kelvin_to_snTTT ( aux, s->val ) )

687/*!
688 \fn int temp_parse_x12 ( struct temp_chunks *t, struct bufr2tac_subset_state *s )
689 \brief Parse a expanded descriptor with X = 12
690 \param t pointer to a struct \ref temp_chunks where to set the results
691 \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
692
693 It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
694*/
695int temp_parse_x12 ( struct temp_chunks *t, struct bufr2tac_subset_state *s )
696{
697 if ( t == NULL )
698 {
699 return 1;
700 }
701
702 switch ( s->a->desc.y )
703 {
704 case 101: // 0 12 101. Temperature/dry-bulb temperature (scale 2)
705 if ( s->rep > 0 && s->r->n > 0 )
706 {
707 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
708 {
709 s->r->raw[s->r->n - 1].T = MISSING_REAL;
710 }
711 else
712 {
713 s->r->raw[s->r->n - 1].T = s->val;
714 }
715 }
716 break;
717
718 case 103: // 0 12 103. Dew-point temperature (scale 2)
719 if ( s->rep > 0 && s->r->n > 0 )
720 {
721 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
722 {
723 s->r->raw[s->r->n - 1].Td = MISSING_REAL;
724 }
725 else
726 {
727 s->r->raw[s->r->n - 1].Td = s->val;
728 }
729 }
730 break;
731
732 default:
733 break;
734 }
735
