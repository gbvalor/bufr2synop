26/*!
27 \fn int get_unexpanded_descriptor_array_from_sec3 ( struct bufr_sequence *s, struct bufrdeco *b )
28 \brief Set to a struct \ref bufr_sequence an unexpanded descriptor array from sec3 in a BUFR report
29 \param s Pointer to the target struct \ref bufr_sequence
30 \param b Pointer to the base struct \ref bufrdeco
31
32 Here we get a struct \ref bufr_sequence from the data about descriptors stored in sec3 of a bufr report
33*/
34int get_unexpanded_descriptor_array_from_sec3 ( struct bufr_sequence *s, struct bufrdeco *b )
35{
36 buf_t i;
37
38 bufrdeco_assert (b != NULL && s != NULL);
39
40 // First we copy the array descriptors in sec3 as level0
41 for ( i = 0; i < b->sec3.ndesc ; i++ )
42 {
43 memcpy ( & ( s->lseq[i] ), & ( b->sec3.unexpanded[i] ), sizeof ( struct bufr_descriptor ) );
44 }
45 s->ndesc = b->sec3.ndesc;
46 return 0;
47}

50/*!
51 \fn int bufrdeco_parse_tree_recursive ( struct bufrdeco *b, struct bufr_sequence *father, const char *key )
52 \brief Parse the descriptor tree in a recursive way
53 \param key string with descriptor in form 'FXXYYY'
54 \param father pointer to the father struct \ref bufr_sequence
55 \param b pointer to the base struct \ref bufrdeco
56
57 Returns 0 if success, 1 otherwise
58 */
59int bufrdeco_parse_tree_recursive ( struct bufrdeco *b, struct bufr_sequence *father, const char *key )
60{
61 buf_t i, j, nl;
62 struct bufr_sequence *l;
63 struct bufr_descriptor *d;
64
65 //bufrdeco_assert (b != NULL);
66
67 if ( key == NULL )
68 {
69 // case first layer
70 memset ( b->tree, 0, sizeof ( struct bufrdeco_expanded_tree ) ); //reset memory
71 b->tree->nseq = 1; // Set current number of sequences in tree, i.e. 1
72 l = & ( b->tree->seq[0] ); // This is to write easily
73 strcpy ( l->key, "000000" ); // Key '000000' is the first descriptor of first sequence of level 0
74 l->level = 0; // Level 0
75 l->father = NULL; // This layer is God, it has not father
76 l->iseq = 0; // first
77 strcpy ( l->name, "Main sequence from SEC3" );
78 // here we get l->ndesc and l->lsec[] array
79 if ( get_unexpanded_descriptor_array_from_sec3 ( l, b ) )
80 {
81 return 1;
82 }
83 }
84 else
85 {
86 // First increase nseq counter
87 if ( b->tree->nseq < NMAXSEQ_DESCRIPTORS )
88 {
89 ( b->tree->nseq ) ++;
90 }
91 else
92 {
93 // No more bufr_sequence permitted
94 snprintf ( b->error, sizeof (b->error),"%s(): Reached max number of bufr_sequence. "
95 "Use bigger NMAXSEQ_LAYER \n" , __func__);
96 return 1;
97 }
98 nl = b->tree->nseq; // To write code easily
99 l = & ( b->tree->seq[nl - 1] ); // To write code easily
100 strcpy ( l->key, key ); // Set the key of sequence in table d (f == 3)
101 l->level = father->level + 1; // level for sequence
102 l->father = father; // set the father
103 l->iseq = nl - 1; // index of sequence in tree
104
105 // here we get ndesc and lsec array from table d
106 if ( bufrdeco_tableD_get_descriptors_array ( l, b, key ) )
107 {
108 return 1; // something went wrong
109 }
110 }
111
112 // Set init replication level values from father, copying the one in replicated[0]
113 for ( i = 1 ; i < l->ndesc ; i++)
114 {
115 l->replicated[i] = l->replicated[0];
116 }
117
118 // Checks for replication levels, no data present, event or conditioning event
119 for ( i = 0; i < l->ndesc ; i++ )
120 {
121 d = & ( l->lseq[i] ); // To write/read code easily
122
123 // Case of replication
124 if ( d->f == 1 )
125 {
126 for ( j = 0; j < d->x; j++ )
127 {
128 if ( d->y == 0 )
129 l->replicated[j + i + 2]++; // a delayed replicator follow
130 else
131 l->replicated[j + i + 1]++; // No delayed descriptor
132 }
133 }
134
135
136 if ( d->f == 2 )
137 {
138 // check for no data present descriptors
139 if ( d->x == 21 &&
140 d->y > 0 )
141 {
142 if ( ( d->y + i ) < l->ndesc )
143 {
144 l->no_data_present.active = 1;
145 l->no_data_present.first = i + 1;
146 l->no_data_present.last = i + d->y;
147
148 }
149 else
150 {
151 // range of no data present out of sequence
152 snprintf ( b->error, sizeof (b->error),"%s(): Range of 'no data present' out of sequence limits\n" , __func__);
153 return 1;
154 }
155 }
156
157 // Check for event descriptors
158 if ( d->x == 41 )
159 {
160 if ( d->y == 0 && ( i + 1 ) < l->ndesc )
161 {
162 l->event.active = 1;
163 l->event.first = i + 1;
164 l->event.last = l->ndesc - 1; // at the moment mark upto final sequence
165 }
166 else if ( ( d->y == 255 ) && l->event.active )
167 {
168 l->event.last = i;
169 }
170 else
171 {
172 snprintf ( b->error, sizeof (b->error),"%s(): unexpected or wrong event descriptor\n" , __func__);
173 return 1;
174 }
175 }
176
177 // Check for conditioning event descriptors
178 if ( d->x == 42 )
179 {
180 if ( d->y == 0 && ( i + 1 ) < l->ndesc )
181 {
182 l->cond_event.active = 1;
183 l->cond_event.first = i + 1;
184 l->cond_event.last = l->ndesc - 1; // at the moment mark upto final sequence
185 }
186 else if ( ( d->y == 255 ) && l->cond_event.active )
187 {
188 l->cond_event.last = i;
189 }
190 else
191 {
192 snprintf ( b->error, sizeof (b->error),"%s(): unexpected or wrong conditioning event event descriptor\n" , __func__);
193 return 1;
194 }
195 }
196
197 // Check for categorical forecasts descriptors
198 if ( d->x == 43 )
199 {
200 if ( d->y == 0 && ( i + 1 ) < l->ndesc )
201 {
202 l->cat_forecast.active = 1;
203 l->cat_forecast.first = i + 1;
204 l->cat_forecast.last = l->ndesc - 1; // at the moment mark upto final sequence
205 }
206 else if ( ( d->y == 255 ) && l->cat_forecast.active )
207 {
208 l->cat_forecast.last = i;
209 }
210 else
211 {
212 snprintf ( b->error, sizeof (b->error),"%s(): unexpected or wrong categorical forecasts descriptor\n" , __func__);
213 return 1;
214 }
215 }
216 }
217 }
218
219 // now we detect sons and go to parse them
220 for ( i = 0; i < l->ndesc ; i++ )
221 {
222 // we search for descriptors with f == 3
223 if ( l->lseq[i].f != 3 )
224 {
225 l->sons[i] = NULL;
226 continue;
227 }
228 l->sons[i] = & ( b->tree->seq[b->tree->nseq] );
229 l->sons[i]->replicated[0] = l->replicated[i]; // This is a trick to pass replication level to sons
230
231 // we then recursively parse the son
232 if ( bufrdeco_parse_tree_recursive ( b, l, l->lseq[i].c ) )
233 {
234 return 1;
235 }
236 }
237
238 // if we are here all gone well
239 return 0;
240}
