26/*!
27 \fn char * grad_to_ec ( char *target, double grad )
28 \brief Converts elevation in grads to ec (code table 1004)
29 \param grad the elevation angle (degrees)
30 \param target the resulting code
31*/
32char * grad_to_ec ( char *target, double grad )
33{
34 if ( grad < 5.0 )
35 strcpy ( target, "9" );
36 else if ( grad < 6.5 )
37 strcpy ( target, "8" );
38 else if ( grad < 8.0 )
39 strcpy ( target, "7" );
40 else if ( grad < 10.5 )
41 strcpy ( target, "6" );
42 else if ( grad < 13.5 )
43 strcpy ( target, "5" );
44 else if ( grad < 17.5 )
45 strcpy ( target, "4" );
46 else if ( grad < 25.0 )
47 strcpy ( target, "3" );
48 else if ( grad < 40.0 )
49 strcpy ( target, "2" );
50 else if ( grad >= 45.0 )
51 strcpy ( target, "1" );
52 else
53 strcpy ( target, "0" );
54 return target;
55}

57/*!
58 \fn int syn_parse_x07 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
59 \brief Parse a expanded descriptor with X = 07
60 \param syn pointer to a struct \ref synop_chunks where to set the results
61 \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
62
63 It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
64*/
65int syn_parse_x07 ( struct synop_chunks *syn, struct bufr2tac_subset_state *s )
66{
67 // this is to avoid warning
68 if ( syn == NULL )
69 {
70 return 1;
71 }
72
73 switch ( s->a->desc.y )
74 {
75 case 1: // 0 07 001 . Heigh of station
76 case 30: // 0 07 030 . Height of station ground above msl
77 case 31: // 0 07 031 . Height of barometer above msl
78 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
79 {
80 return 0;
81 }
82 if ( syn->s0.h0h0h0h0[0] == 0 )
83 {
84 if (s->ival > 9999 || s->ival < -9999)
85 return 1;
86 sprintf ( syn->s0.h0h0h0h0, "%04d", s->ival );
87 syn->s0.im[0] = '1';// set unit as m
88 s->mask |= SUBSET_MASK_HAVE_ALTITUDE;
89 }
90 s->alt = s->val;
91 break;
92 case 4: // 0 07 004 . Pressure at standard level
93 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
94 {
95 return 0;
96 }
97 if ( s->ival == 100000 )
98 {
99 strcpy ( syn->s1.a3, "1" );
100 }
101 else if ( s->ival == 92500 )
102 {
103 strcpy ( syn->s1.a3, "2" );
104 }
105 else if ( s->ival == 85000 )
106 {
107 strcpy ( syn->s1.a3, "8" );
108 }
109 else if ( s->ival == 70000 )
110 {
111 strcpy ( syn->s1.a3, "7" );
112 }
113 else if ( s->ival == 50000 )
114 {
115 strcpy ( syn->s1.a3, "5" );
116 }
117 break;
118
119 case 21: // 0 07 021 . Elevation angle
120 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
121 {
122 return 0;
123 }
124 grad_to_ec ( syn->s3.ec, s->val );
125 syn->mask |= SYNOP_SEC3;
126 break;
127
128 case 32: // 0 07 032 Height of sensor above local ground
129 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
130 {
131 s->hsensor = -999.9; // clean value
132 return 0;
133 }
134 s->hsensor = s->val;
135 break;
136
137 case 33: // 0 07 033 Height of sensor above water surface
138 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
139 {
140 s->hwsensor = -999.9; // clean value
141 return 0;
142 }
143 s->hwsensor = s->val;
144 break;
145
146 default:
147 break;
148 }
149
150 return 0;
151}

153/*!
154 \fn int buoy_parse_x07 ( struct buoy_chunks *b, struct bufr2tac_subset_state *s )
155 \brief Parse a expanded descriptor with X = 07
156 \param b pointer to a struct \ref buoy_chunks where to set the results
157 \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
158
159 It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
160*/
161int buoy_parse_x07 ( struct buoy_chunks *b, struct bufr2tac_subset_state *s )
162{
163
164 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
165 {
166 return 0;
167 }
168
169 if ( b == NULL )
170 {
171 return 1;
172 }
173
174 switch ( s->a->desc.y )
175 {
176 case 62: // 0 07 062 Depth below sea/water surface
177 case 63: // 0 07 063 Profundity below sea/water level
178 if ( s->k_rep == ( s->i - 1 ) ) // Case of first layer after a replicator
179 {
180 s->layer = 0;
181 }
182 else
183 {
184 ( s->layer ) ++;
185 }
186 s->deep = ( int ) ( s->val + 0.5 );
187 break;
188 default:
189 break;
190 }
191 return 0;
192}

237/*!
238 \fn int temp_parse_x07 ( struct synop_chunks *c, struct bufr2tac_subset_state *s )
239 \brief Parse a expanded descriptor with X = 07
240 \param c pointer to a struct \ref temp_chunks where to set the results
241 \param s pointer to a struct \ref bufr2tac_subset_state where is stored needed information in sequential analysis
242
243 It returns 0 if success, 1 if problems when processing. If a descriptor is not processed returns 0 anyway
244*/
245int temp_parse_x07 ( struct temp_chunks *t, struct bufr2tac_subset_state *s )
246{
247 if ( s->a->mask & DESCRIPTOR_VALUE_MISSING )
248 {
249 return 0;
250 }
251
252// this is to avoid warning
253 if ( t == NULL )
254 {
255 return 1;
256 }
257
258 switch ( s->a->desc.y )
259 {
260 case 4: // 0 07 004 . Pressure
261 if ( s->rep > 0 && s->r->n > 0 )
262 {
263 s->r->raw[s->r->n - 1].p = s->val;
264 }
265 else if ( s->w->n > 0 )
266 {
267 s->w->raw[s->w->n - 1].p = s->val;
268 }
269 break;
270
271 case 30: // 0 07 030 . Height of station ground above msl
272 sprintf ( t->a.s1.h0h0h0h0, "%04.0lf" , s->val );
273 sprintf ( t->b.s1.h0h0h0h0, "%04.0lf" , s->val );
274 sprintf ( t->c.s1.h0h0h0h0, "%04.0lf" , s->val );
275 sprintf ( t->d.s1.h0h0h0h0, "%04.0lf" , s->val );
276 break;
277
278 case 31: // 0 07 031 . Height of barometer above msl
279 s->alt = s->val;
280 break;
281
282 default:
283 break;
284 }
285
286 return 0;
