28 \brief Init a struct \ref bufr_tables allocating space
29 \param t pointer to the target pointer to struct \ref bufr_tables
30
31 Returns 0 if succeeded, 1 otherwise

33int bufrdeco_init_tables ( struct bufr_tables **t )
34{
35 bufrdeco_assert ( t != NULL );
36
37 if ( *t != NULL )
38 free ( ( void * ) *t );
39
40 if ( ( *t = ( struct bufr_tables * ) calloc ( 1, sizeof ( struct bufr_tables ) ) ) == NULL )

42 return 0;
43}

103/*!
104 \fn int bufrdeco_substitute_tables ( struct bufr_tables **replaced, struct bufr_tables *source, struct bufrdeco *b )
105 \brief substitute an struct \ref bufr_tables into a struct \ref bufrdeco
106 \param replaced Pointer where to set the replaced pointer
107 \param source Pointer to a struct \ref bufr_tables
108 \param b pointer to the container basic struct \ref bufrdeco
109
110 Remember that the struct \ref bufr_tables used in bufrdeco library is the one which pointer is in struct
111 \ref bufrdeco . To avoid problems the struct must be initialized before substituted in this fucntion.
112 Both source and replaced structs are not modified.
113
114 This is useful if we do not want to read and parse tables again if the caller has a pool of
115 already readed tables.
116
117 Returns 0 if succeeded, 1 otherwise
118*/
119int bufrdeco_substitute_tables ( struct bufr_tables **replaced, struct bufr_tables *source, struct bufrdeco *b )
120{
121 bufrdeco_assert ( b != NULL );
122
123 *replaced = b->tables;
124 if ( source == NULL )
125 {
126 // allocate memory for table
127 return bufrdeco_init_tables ( & ( b->tables ) );
128 }
129 else
130 b->tables = source;
131 return 0;
132}

135 \fn int bufrdeco_init_subset_sequence_data ( struct bufrdeco_subset_sequence_data *ba )
136 \brief Init a struct \ref bufrdeco_subset_sequence_data

139 It is supossed that no memory is allocated for sequence. If we are not sure better use
140 function \ref bufrdeco_clean_subset_sequence_data

145{
146 bufrdeco_assert ( ba != NULL );
147
148 memset ( ba, 0, sizeof ( struct bufrdeco_subset_sequence_data ) );
149 if ( ( ba->sequence = ( struct bufr_atom_data * ) calloc ( 1, BUFR_NMAXSEQ * sizeof ( struct bufr_atom_data ) ) ) == NULL )
150 {
151 fprintf ( stderr,"%s():Cannot allocate memory for atom data array\n", __func__ );
152 return 1;
153 }
154 ba->dim = BUFR_NMAXSEQ;
155 return 0;

199/*!
200 \fn int bufrdeco_init_compressed_data_references ( struct bufrdeco_compressed_data_references *rf )
201 \brief Init a struct bufrdeco_compressed_data_references
202 \param rf pointer ti the target struct
203
204 If already memory is allocated for array of references then just adjust the used index to zero. Otherwise
205 it allocate the needed memory and init the struct
206
207 If succeeded return 0, otherwise 1
208*/
209int bufrdeco_init_compressed_data_references ( struct bufrdeco_compressed_data_references *rf )
210{
211 bufrdeco_assert ( rf != NULL );
212
213 if ( rf->refs != NULL && rf->dim != 0 )
214 {
215 rf->nd = 0; // Here we set the used elements to 0 of dim
216 }
217 else if ( rf->refs == NULL )
218 {
219 // Here memory is still not allocated. Proceed to allocate with BUFR_NMAXSEQ
220 if ( ( rf->refs = ( struct bufrdeco_compressed_ref * ) calloc ( 1, BUFR_NMAXSEQ * sizeof ( struct bufrdeco_compressed_ref ) ) ) == NULL )
221 {
222 fprintf ( stderr,"%s(): Cannot allocate memory for bufrdeco_compressed_ref array\n", __func__ );
223 return 1;
224 }
225 rf->nd = 0; // Set de used elements (bufrdeco_compressed_ref)
226 rf->dim = BUFR_NMAXSEQ; // Set de allocated bufr_compressed_rer elements
227 }
228 return 0;
229}

267/*!
268 * \fn int bufrdeco_allocate_bitmap ( struct bufrdeco *b )
269 * \brief allocate bitmap
270 *
271 * If succeeded return 0, otherwise 1
272 */
273int bufrdeco_allocate_bitmap ( struct bufrdeco *b )
274{
275 buf_t nba;
276
277 bufrdeco_assert ( b != NULL );
278
279 nba = b->bitmap.nba;
280
281 if ( nba < BUFR_MAX_BITMAPS )
282 {
283 if ( b->bitmap.bmap[nba] != NULL )
284 {
285 // the bitmap already is allocated
286 return 0;
287 }
288 // let's try to allocate it!
289 if ( ( b->bitmap.bmap[nba] = ( struct bufrdeco_bitmap * ) calloc ( 1, sizeof ( struct bufrdeco_bitmap ) ) ) == NULL )
290 {
291 snprintf ( b->error, sizeof ( b->error ),"%s(): Cannot allocate space for struct bufrdeco_bitmap\n", __func__ );
292 return 1;
293 }
294 // Update de counter
295 ( b->bitmap.nba )++;
296 return 0;
297 }
298 else
299 {
300 snprintf ( b->error, sizeof ( b->error ),"%s(): Too much bitmaps to allocate. The limit is %d\n", __func__, BUFR_MAX_BITMAPS );
301 return 1;
302 }
303}

306/*!
307 * \fn int bufrdeco_clean_bitmaps ( struct bufrdeco *b )
308 * \brief Clean all allocated bitmaps, but still is in memory
309 *
310 * If succeeded return 0, otherwise 1
311 */
312int bufrdeco_clean_bitmaps ( struct bufrdeco *b )
313{
314 buf_t i;
315
316 bufrdeco_assert ( b != NULL );
317
318 for ( i = 0; i < b->bitmap.nba ; i++ )
319 {
320 if ( b->bitmap.bmap[i] == NULL )
321 continue;
322 memset ( b->bitmap.bmap[i], 0, sizeof ( struct bufrdeco_bitmap ) );
323 }
324 b->bitmap.nba = 0;
325 return 0;
326}
